17  package com.google.common.util.concurrent;
18  
19  import com.google.common.annotations.Beta;
20  import com.google.common.annotations.GwtCompatible;
21  
22  import java.util.concurrent.ExecutionException;
23  import java.util.concurrent.Future;
24  
25  /**
26   * Provides a backup {@code Future} to replace an earlier failed {@code Future}. An implementation
27   * of this interface can be applied to an input {@code Future} with {@link Futures#withFallback}.
28   *
29   * @param <V> the result type of the provided backup {@code Future}
30   *
31   * @author Bruno Diniz
32   * @since 14.0
33   * @deprecated This interface's main user, {@link Futures#withFallback(ListenableFuture,
34   *     FutureFallback) Futures.withFallback}, has been updated to use {@link AsyncFunction}. We
35   *     recommend that other APIs be updated in the same way. This interface will be removed in Guava
36   *     release 20.0.
37   */
38  @Beta
39  @Deprecated
40  @GwtCompatible
41  public interface FutureFallback<V> {
42    /**
43     * Returns a {@code Future} to be used in place of the {@code Future} that failed with the given
44     * exception. The exception is provided so that the {@code Fallback} implementation can
45     * conditionally determine whether to propagate the exception or to attempt to recover.
46     *
47     * @param t the exception that made the future fail. If the future's {@link Future#get() get}
48     *     method throws an {@link ExecutionException}, then the cause is passed to this method. Any
49     *     other thrown object is passed unaltered.
50     */
51    ListenableFuture<V> create(Throwable t) throws Exception;
52  }
