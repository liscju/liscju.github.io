112     /** Handler for visibility event. */
113     private static class DihedralSunVisiHandler implements EventHandler<DihedralFieldOfViewDetector> {
114 
115         public Action eventOccurred(final SpacecraftState s, final DihedralFieldOfViewDetector detector,
116                                     final boolean increasing)
117             throws OrekitException {
118             if (increasing) {
119                 //System.err.println(" Sun visibility starts " + s.getDate());
120                 AbsoluteDate startVisiDate = new AbsoluteDate(new DateComponents(1969, 8, 28),
121                                                               new TimeComponents(1, 19, 00.381),
122                                                               TimeScalesFactory.getUTC());
123 
124                 Assert.assertTrue(s.getDate().durationFrom(startVisiDate) <= 1);
125                 return Action.CONTINUE;
126             } else {
127                 AbsoluteDate endVisiDate = new AbsoluteDate(new DateComponents(1969, 8, 28),
128                                                               new TimeComponents(1, 39 , 42.674),
129                                                               TimeScalesFactory.getUTC());
130                 Assert.assertTrue(s.getDate().durationFrom(endVisiDate) <= 1);
131                 //System.err.println(" Sun visibility ends at " + s.getDate());
132                 return Action.CONTINUE;//STOP;
133             }
134         }
135 
136         public SpacecraftState resetState(DihedralFieldOfViewDetector detector, SpacecraftState oldState) {
137             return oldState;
138         }
