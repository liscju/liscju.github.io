260     private PVCoordinates propagateInType(SpacecraftState state, double dP,
261                                           OrbitType type, PositionAngle angle)
262         throws PropagationException {
263 
264         final double dt = 3200;
265         final double minStep = 0.001;
266         final double maxStep = 1000;
267 
268         double[][] tol = NumericalPropagator.tolerances(dP, state.getOrbit(), type);
269         AdaptiveStepsizeIntegrator integrator =
270                 new DormandPrince853Integrator(minStep, maxStep, tol[0], tol[1]);
271         NumericalPropagator newPropagator = new NumericalPropagator(integrator);
272         newPropagator.setOrbitType(type);
273         newPropagator.setPositionAngleType(angle);
274         newPropagator.setInitialState(state);
275         for (ForceModel force: propagator.getForceModels()) {
276             newPropagator.addForceModel(force);
277         }
278         return newPropagator.propagate(state.getDate().shiftedBy(dt)).getPVCoordinates();
279 
280     }

668     private static class CheckingHandler<T extends EventDetector> implements EventHandler<T> {
669 
670         private final Action actionOnEvent;
671         private boolean gotHere;
672 
673         public CheckingHandler(final Action actionOnEvent) {
674             this.actionOnEvent = actionOnEvent;
675             this.gotHere       = false;
676         }
677 
678         public void assertEvent(boolean expected) {
679             Assert.assertEquals(expected, gotHere);
680         }
681 
682         public Action eventOccurred(SpacecraftState s, T detector, boolean increasing) {
683             gotHere = true;
684             return actionOnEvent;
685         }
686 
687         public SpacecraftState resetState(T detector, SpacecraftState oldState)
688             throws OrekitException {
689             return oldState;
690         }
