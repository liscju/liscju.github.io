24      @Test
25      public void testTrivial() throws Exception {
26          final AbsoluteDate date = new AbsoluteDate("2011-12-12T11:57:20.000", TimeScalesFactory.getUTC());
27          final Orbit orbit1 = new CircularOrbit(7204535.848109436, -4.484755873986251E-4, 0.0011562979012178316,
28                                                 Math.toRadians(98.74341600466741), Math.toRadians(43.32990110790338),
29                                                 Math.toRadians(180.0), PositionAngle.MEAN, FramesFactory.getGCRF(),
30                                                 date, Constants.WGS84_EARTH_MU);
31          final SpacecraftState initialState = new SpacecraftState(orbit1);
32          // Set up the numerical propagator
33          final double[][] tol = NumericalPropagator.tolerances(1.0, initialState.getOrbit(), initialState.getOrbit().getType());
34          final double minStep = 1.;
35          final double maxStep = 200.;
36          AdaptiveStepsizeIntegrator integrator = new DormandPrince853Integrator(minStep, maxStep, tol[0], tol[1]);
37          integrator.setInitialStepSize(100.);
38          final NumericalPropagator prop = new NumericalPropagator(integrator);
39          prop.setInitialState(initialState);
40  
41          final OsculatingToMeanElementsConverter converter = new OsculatingToMeanElementsConverter(initialState, 2, prop);
42          final SpacecraftState meanOrbit = converter.convert();
43  
44          final double eps  = 1.e-15;
45  
46          Assert.assertEquals(orbit1.getA(), meanOrbit.getA(), eps * orbit1.getA());
47          Assert.assertEquals(orbit1.getEquinoctialEx(), meanOrbit.getEquinoctialEx(), eps);
48          Assert.assertEquals(orbit1.getEquinoctialEy(), meanOrbit.getEquinoctialEy(), eps);
49          Assert.assertEquals(orbit1.getHx(), meanOrbit.getHx(), eps);
50          Assert.assertEquals(orbit1.getHy(), meanOrbit.getHy(), eps);
51          Assert.assertEquals(MathUtils.normalizeAngle(orbit1.getLM(), FastMath.PI),
52                              MathUtils.normalizeAngle(meanOrbit.getLM(), FastMath.PI), eps);
53      }
54

56      public void setUp() throws Exception {
57          Utils.setDataRoot("regular-data:potential/shm-format");
58      }
