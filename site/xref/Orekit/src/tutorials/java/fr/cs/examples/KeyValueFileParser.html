40  /** Simple parser for key/value files.
41   * @param Key type of the parameter keys
42   */
43  public class KeyValueFileParser<Key extends Enum<Key>> {
44  
45      /** Error message for unknown frame. */
46      private static final Localizable UNKNOWN_FRAME =
47          new DummyLocalizable("unknown frame {0}");
48  
49      /** Error message for not Earth frame. */
50      private static final Localizable NOT_EARTH_FRAME =
51          new DummyLocalizable("frame {0} is not an Earth frame");
52  
53      /** Enum type. */
54      private final Class<Key> enumType;
55  
56      /** Key/value map. */
57      private final Map<Key, String> map;
58  
59      /** Simple constructor.
60       * @param enumType type of the parameters keys enumerate
61       */
62      public KeyValueFileParser(Class<Key> enumType) {
63          this.enumType = enumType;
64          map = new HashMap<Key, String>();
65      }
66  
67      /** Parse an input file.
68       * <p>
69       * The input file syntax is a set of key=value lines. Blank lines and lines
70       * starting with '#' (after whitespace trimming) are silently ignored. The
71       * equal sign may be surrounded by space characters. Keys must correspond to
72       * the {@link Key} enumerate constants, given that matching is not
73       * case sensitive and that '_' characters may appear as '.' characters in the
74       * file. This means that the lines:
75       * <pre>
76       *   # this is the semi-major axis
77       *   orbit.circular.a   = 7231582
78       * </pre>
79       * are perfectly right and correspond to key {@code Key#ORBIT_CIRCULAR_A} if
80       * such a constant exists in the enumerate.
81       * </p>
82       * @param input input stream
83       * @exception IOException if input file cannot be read
84       * @exception IllegalArgumentException if a line cannot be read properly
85       */
86      public void parseInput(final InputStream input) throws IOException, IllegalArgumentException {
87  
88          BufferedReader reader = new BufferedReader(new InputStreamReader(input));
89          for (String line = reader.readLine(); line != null; line = reader.readLine()) {
90              line = line.trim();
91              // we ignore blank lines and line starting with '#'
92              if ((line.length() > 0) && !line.startsWith("#")) {
93                  String[] fields = line.split("\\s*=\\s*");
94                  if (fields.length != 2) {
95                      throw new IllegalArgumentException(line);
96                  }
97                  Key key = Key.valueOf(enumType, fields[0].toUpperCase().replaceAll("\\.", "_"));
98                  map.put(key, fields[1]);
99              }
100         }
101         reader.close();
102 
103     }
104 
105     /** Check if a key is contained in the map.
106      * @param key parameter key
107      * @return true if the key is contained in the map
108      */
109     public boolean containsKey(final Key key) {
110         return map.containsKey(key);
111     }
112 
113     /** Get a raw string value from a parameters map.
114      * @param key parameter key
115      * @return string value corresponding to the key
116      * @exception NoSuchElementException if key is not in the map
117      */
118     public String getString(final Key key) throws NoSuchElementException {
119         final String value = map.get(key);
120         if (value == null) {
121             throw new NoSuchElementException(key.toString());
122         }
123         return value.trim();
124     }
125 
126     /** Get a raw double value from a parameters map.
127      * @param key parameter key
128      * @return double value corresponding to the key
129      * @exception NoSuchElementException if key is not in the map
130      */
131     public double getDouble(final Key key) throws NoSuchElementException {
132         return Double.parseDouble(getString(key));
133     }
134 
135     /** Get a raw int value from a parameters map.
136      * @param key parameter key
137      * @return int value corresponding to the key
138      * @exception NoSuchElementException if key is not in the map
139      */
140     public int getInt(final Key key) throws NoSuchElementException {
141         return Integer.parseInt(getString(key));
142     }
143 
144     /** Get a raw boolean value from a parameters map.
145      * @param key parameter key
146      * @return boolean value corresponding to the key
147      * @exception NoSuchElementException if key is not in the map
148      */
149     public boolean getBoolean(final Key key) throws NoSuchElementException {
150         return Boolean.parseBoolean(getString(key));
151     }
152 
153     /** Get an angle value from a parameters map.
154      * <p>
155      * The angle is considered to be in degrees in the file, it will be returned in radians
156      * </p>
157      * @param key parameter key
158      * @return angular value corresponding to the key, in radians
159      * @exception NoSuchElementException if key is not in the map
160      */
161     public double getAngle(final Key key) throws NoSuchElementException {
162         return FastMath.toRadians(getDouble(key));
163     }
164 
165     /** Get a date value from a parameters map.
166      * <p>
167      * The date is considered to be in UTC in the file
168      * </p>
169      * @param key parameter key
170      * @param scale time scale in which the date is to be parsed
171      * @return date value corresponding to the key
172      * @exception NoSuchElementException if key is not in the map
173      */
174     public AbsoluteDate getDate(final Key key, TimeScale scale) throws NoSuchElementException {
175         return new AbsoluteDate(getString(key), scale);
176     }
177 
178     /** Get a time value from a parameters map.
179      * @param key parameter key
180      * @return time value corresponding to the key
181      * @exception NoSuchElementException if key is not in the map
182      */
183     public TimeComponents getTime(final Key key) throws NoSuchElementException {
184         return TimeComponents.parseTime(getString(key));
185     }
186 
187     /** Get a vector value from a parameters map.
188      * @param xKey parameter key for abscissa
189      * @param yKey parameter key for ordinate
190      * @param zKey parameter key for height
191      * @param scale time scale in which the date is to be parsed
192      * @return date value corresponding to the key
193      * @exception NoSuchElementException if key is not in the map
194      */
195     public Vector3D getVector(final Key xKey, final Key yKey, final Key zKey)
196         throws NoSuchElementException {
197         return new Vector3D(getDouble(xKey), getDouble(yKey), getDouble(zKey));
198     }
199 
200     /** Get an inertial frame from a parameters map.
201      * @param key parameter key
202      * @return inertial frame corresponding to the key
203      * @exception NoSuchElementException if key is not in the map
204      * @exception OrekitException if frame cannot be built
205      */
206     public Frame getInertialFrame(final Key key) throws NoSuchElementException, OrekitException {
207 
208         // get the name of the desired frame
209         final String frameName = getString(key);
210 
211         // check the name against predefined frames
212         for (Predefined predefined : Predefined.values()) {
213             if (frameName.equals(predefined.getName())) {
214                 if (FramesFactory.getFrame(predefined).isPseudoInertial()) {
215                     return FramesFactory.getFrame(predefined);
216                 } else {
217                     throw new OrekitException(OrekitMessages.NON_PSEUDO_INERTIAL_FRAME,
218                                               frameName);
219                 }
220             }
221         }
222 
223         // none of the frames match the name
224         throw new OrekitException(UNKNOWN_FRAME, frameName);
225 
226     }
227 
228     /** Get an Earth frame from a parameters map.
229      * <p>
230      * We consider Earth frames are the frames with name starting with "ITRF".
231      * </p>
232      * @param key parameter key
233      * @param parameters key/value map containing the parameters
234      * @return Earth frame corresponding to the key
235      * @exception NoSuchElementException if key is not in the map
236      * @exception OrekitException if frame cannot be built
237      */
238     public Frame getEarthFrame(final Key key)
239             throws NoSuchElementException, OrekitException {
240 
241         // get the name of the desired frame
242         final String frameName = getString(key);
243 
244         // check the name against predefined frames
245         for (Predefined predefined : Predefined.values()) {
246             if (frameName.equals(predefined.getName())) {
247                 if (frameName.startsWith("ITRF")) {
248                     return FramesFactory.getFrame(predefined);
249                 } else {
250                     throw new OrekitException(NOT_EARTH_FRAME, frameName);
251                 }
252             }
253         }
254 
255         // none of the frames match the name
256         throw new OrekitException(UNKNOWN_FRAME, frameName);
257 
258     }
