17  package org.orekit.propagation.conversion;
18  
19  import java.util.ArrayList;
20  import java.util.Collections;
21  import java.util.List;
22  
23  import org.apache.commons.math3.exception.util.LocalizedFormats;
24  import org.orekit.errors.OrekitIllegalArgumentException;
25  import org.orekit.errors.OrekitMessages;
26  import org.orekit.forces.gravity.NewtonianAttraction;
27  import org.orekit.frames.Frame;
28  import org.orekit.orbits.Orbit;
29  import org.orekit.orbits.OrbitType;
30  import org.orekit.orbits.PositionAngle;
31  import org.orekit.time.AbsoluteDate;
32  
33  /** Base class for propagator builders.
34   * @author Pascal Parraud
35   * @since 7.1
36   */
37  public abstract class AbstractPropagatorBuilder implements PropagatorBuilder {
38  
39      /** Frame in which the orbit is propagated. */
40      private final Frame frame;
41  
42      /** Central attraction coefficient (m³/s²). */
43      private double mu;
44  
45      /** List of the supported parameters names. */
46      private List<String> supportedParameters;
47  
48      /** List of the free parameters names. */
49      private List<String> freeParameters;
50  
51      /** Orbit type to use. */
52      private final OrbitType orbitType;
53  
54      /** Position angle type to use. */
55      private final PositionAngle positionAngle;
56  
57      /** Build a new instance.
58       * @param frame the frame in which the orbit is propagated
59       *        (<em>must</em> be a {@link Frame#isPseudoInertial pseudo-inertial frame})
60       * @param mu central attraction coefficient (m³/s²)
61       * @param orbitType orbit type to use
62       * @param positionAngle position angle type to use
63       * @since 7.1
64       */
65      public AbstractPropagatorBuilder(final Frame frame, final double mu,
66                                       final OrbitType orbitType, final PositionAngle positionAngle) {
67          this.frame               = frame;
68          this.mu                  = mu;
69          this.supportedParameters = new ArrayList<String>();
70          this.freeParameters      = new ArrayList<String>();
71          this.orbitType           = orbitType;
72          this.positionAngle       = positionAngle;
73          addSupportedParameter(NewtonianAttraction.CENTRAL_ATTRACTION_COEFFICIENT);
74      }
75  
76      /** {@inheritDoc} */
77      public OrbitType getOrbitType() {
78          return orbitType;
79      }
80  
81      /** {@inheritDoc} */
82      public PositionAngle getPositionAngle() {
83          return positionAngle;
84      }
85  
86      /** {@inheritDoc} */
87      public Frame getFrame() {
88          return frame;
89      }
90  
91      /** {@inheritDoc} */
92      public List<String> getSupportedParameters() {
93          return Collections.unmodifiableList(supportedParameters);
94      }
95  
96      /** {@inheritDoc} */
97      public void setFreeParameters(final List<String> parameters)
98          throws OrekitIllegalArgumentException {
99          for (String name : parameters) {
100             if (!supportedParameters.contains(name)) {
101                 throw new OrekitIllegalArgumentException(OrekitMessages.UNSUPPORTED_PARAMETER_NAME, name);
102             }
103         }
104         freeParameters = new ArrayList<String>(parameters);
105     }
106 
107     /** {@inheritDoc} */
108     public List<String> getFreeParameters() {
109         return Collections.unmodifiableList(freeParameters);
110     }
111 
112     /** {@inheritDoc}
113      * <p>
114      * The abstract base class only supports {@link
115      * NewtonianAttraction#CENTRAL_ATTRACTION_COEFFICIENT}, specialized propagator
116      * builders may support more parameters.
117      * </p>
118      */
119     public double getParameter(final String name)
120         throws OrekitIllegalArgumentException {
121         if (NewtonianAttraction.CENTRAL_ATTRACTION_COEFFICIENT.equals(name)) {
122             return mu;
123         }
124         throw new OrekitIllegalArgumentException(OrekitMessages.UNSUPPORTED_PARAMETER_NAME, name);
125     }
126 
127     /** {@inheritDoc}
128      * <p>
129      * The abstract base class only supports {@link
130      * NewtonianAttraction#CENTRAL_ATTRACTION_COEFFICIENT}, specialized propagator
131      * builders may support more parameters.
132      * </p>
133      */
134     public void setParameter(final String name, final double value)
135         throws OrekitIllegalArgumentException {
136         if (NewtonianAttraction.CENTRAL_ATTRACTION_COEFFICIENT.equals(name)) {
137             this.mu = value;
138         } else {
139             throw new OrekitIllegalArgumentException(OrekitMessages.UNSUPPORTED_PARAMETER_NAME, name);
140         }
141     }
142 
143     /** Check the size of the parameters array.
144      * @param parameters to configure the propagator
145      * @exception OrekitIllegalArgumentException if the number of
146      * parameters is not 6 (for initial state) plus the number of free
147      * parameters
148      */
149     protected void checkParameters(final double[] parameters)
150         throws OrekitIllegalArgumentException {
151         if (parameters.length != (freeParameters.size() + 6)) {
152             throw new OrekitIllegalArgumentException(LocalizedFormats.DIMENSIONS_MISMATCH_SIMPLE,
153                                                      parameters.length, freeParameters.size() + 6);
154         }
155     }
156 
157     /** Crate the orbit for the first 6 parameters.
158      * @param date date associated to the parameters to configure the initial state
159      * @param parameters set of position/velocity(/free) parameters to configure the propagator
160      * @return initial orbit
161      */
162     protected Orbit createInitialOrbit(final AbsoluteDate date, final double[] parameters) {
163         return getOrbitType().mapArrayToOrbit(parameters, positionAngle, date, mu, frame);
164     }
165 
166     /** Add a supported parameter name.
167      * @param name name of a supported parameter
168      * @exception OrekitIllegalArgumentException if the name is already supported
169      */
170     protected void addSupportedParameter(final String name)
171         throws OrekitIllegalArgumentException {
172         if (supportedParameters.contains(name)) {
173             throw new OrekitIllegalArgumentException(OrekitMessages.DUPLICATED_PARAMETER_NAME, name);
174         }
175         supportedParameters.add(name);
176     }
177 
178 }
