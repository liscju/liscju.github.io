18  
19  import java.util.ArrayList;
20  import java.util.Iterator;
21  import java.util.List;
22  
23  import org.orekit.attitudes.Attitude;
24  import org.orekit.attitudes.AttitudeProvider;
25  import org.orekit.errors.OrekitException;
26  import org.orekit.errors.OrekitIllegalArgumentException;
27  import org.orekit.forces.ForceModel;
28  import org.orekit.forces.gravity.NewtonianAttraction;
29  import org.orekit.frames.Frame;

31  import org.orekit.orbits.OrbitType;
32  import org.orekit.orbits.PositionAngle;

38  /** Builder for numerical propagator.
39   * @author Pascal Parraud
40   * @since 6.0
41   */
42  public class NumericalPropagatorBuilder extends AbstractPropagatorBuilder {
43  
44      /** First order integrator builder for propagation. */
45      private final FirstOrderIntegratorBuilder builder;
46  
47      /** Force models used during the extrapolation of the orbit. */
48      private final List<ForceModel> forceModels;
49  
50      /** Current mass for initial state (kg). */
51      private double mass;
52  
53      /** Attitude provider. */
54      private AttitudeProvider attProvider;
55  
56      /** Build a new instance.
57       * @param mu central attraction coefficient (m³/s²)
58       * @param frame the frame in which the orbit is propagated
59       * (<em>must</em> be a {@link Frame#isPseudoInertial pseudo-inertial frame})
60       * @param builder first order integrator builder
61       * @deprecated as of 7.1, replaced with {@link #NumericalPropagatorBuilder(double,
62       * Frame, FirstOrderIntegratorBuilder, OrbitType, PositionAngle)}
63       */
64      @Deprecated
65      public NumericalPropagatorBuilder(final double mu,
66                                        final Frame frame,
67                                        final FirstOrderIntegratorBuilder builder) {
68          this(mu, frame, builder, OrbitType.CARTESIAN, PositionAngle.TRUE);
69      }
70  
71      /** Build a new instance.
72       * @param mu central attraction coefficient (m³/s²)
73       * @param frame the frame in which the orbit is propagated
74       * (<em>must</em> be a {@link Frame#isPseudoInertial pseudo-inertial frame})
75       * @param builder first order integrator builder
76       * @param orbitType orbit type to use
77       * @param positionAngle position angle type to use
78       * @since 7.1
79       */
80      public NumericalPropagatorBuilder(final double mu,
81                                        final Frame frame,
82                                        final FirstOrderIntegratorBuilder builder,
83                                        final OrbitType orbitType, final PositionAngle positionAngle) {
84          super(frame, mu, orbitType, positionAngle);
85          this.builder     = builder;
86          this.forceModels = new ArrayList<ForceModel>();
87          this.mass        = Propagator.DEFAULT_MASS;
88          this.attProvider = Propagator.DEFAULT_LAW;
89      }
90  
91      /** Set the attitude provider.
92       * @param attitudeProvider attitude provider
93       */
94      public void setAttitudeProvider(final AttitudeProvider attitudeProvider) {
95          this.attProvider = attitudeProvider;
96      }
97  
98      /** Set the initial mass.
99       * @param mass the mass (kg)
100      */
101     public void setMass(final double mass) {
102         this.mass = mass;
103     }
104 
105     /** Add a force model to the global perturbation model.
106      * <p>If this method is not called at all, the integrated orbit will follow
107      * a keplerian evolution only.</p>
108      * @param model perturbing {@link ForceModel} to add
109      */
110     public void addForceModel(final ForceModel model) {
111         forceModels.add(model);
112         for (final String name : model.getParametersNames()) {
113             // add model parameters, taking care of
114             // Newtonian central attraction which is already supported by base class
115             if (NewtonianAttraction.CENTRAL_ATTRACTION_COEFFICIENT.equals(name)) {
116                 super.setParameter(name, model.getParameter(name));
117             } else {
118                 addSupportedParameter(name);
119             }
120         }
121     }
122 
123     /** {@inheritDoc} */
124     public NumericalPropagator buildPropagator(final AbsoluteDate date, final double[] parameters)
125         throws OrekitException {
126 
127         checkParameters(parameters);
128         final Orbit orb = createInitialOrbit(date, parameters);
129 
130         final Attitude attitude = attProvider.getAttitude(orb, date, getFrame());
131 
132         final SpacecraftState state = new SpacecraftState(orb, attitude, mass);
133 
134         final Iterator<String> freeItr = getFreeParameters().iterator();
135         for (int i = 6; i < parameters.length; i++) {
136             final String free = freeItr.next();
137             for (String available : getSupportedParameters()) {
138                 if (free.equals(available)) {
139                     setParameter(free, parameters[i]);
140                 }
141             }
142         }
143 
144         final NumericalPropagator propagator = new NumericalPropagator(builder.buildIntegrator(orb));
145         propagator.setOrbitType(getOrbitType());
146         propagator.setPositionAngleType(getPositionAngle());
147         propagator.setAttitudeProvider(attProvider);
148         for (ForceModel model : forceModels) {
149             propagator.addForceModel(model);
150         }
151         propagator.resetInitialState(state);
152 
153         return propagator;
154     }
155 
156     /** {@inheritDoc} */
157     @Override
158     public double getParameter(final String name)
159         throws OrekitIllegalArgumentException {
160         for (ForceModel model : forceModels) {
161             if (model.isSupported(name)) {
162                 return model.getParameter(name);
163             }
164         }
165         return super.getParameter(name);
166     }
167 
168     /** {@inheritDoc} */
169     @Override
170     public void setParameter(final String name, final double value)
171         throws OrekitIllegalArgumentException {
172         for (ForceModel model : forceModels) {
173             if (model.isSupported(name)) {
174                 model.setParameter(name, value);
175                 if (NewtonianAttraction.CENTRAL_ATTRACTION_COEFFICIENT.equals(name)) {
176                     // the central attraction must be configured
177                     // in both the force model and the base class
178                     super.setParameter(name, value);
179                 }
180                 return;
181             }
182         }
183         super.setParameter(name, value);
184     }
185 
186 }
