17  package org.orekit.propagation.conversion;
18  
19  import org.orekit.errors.OrekitException;
20  import org.orekit.frames.Frame;
21  import org.orekit.orbits.OrbitType;
22  import org.orekit.orbits.PositionAngle;
23  import org.orekit.propagation.Propagator;
24  import org.orekit.propagation.analytical.KeplerianPropagator;
25  import org.orekit.time.AbsoluteDate;
26  
27  /** Builder for Keplerian propagator.
28   * @author Pascal Parraud
29   * @since 6.0
30   */
31  public class KeplerianPropagatorBuilder extends AbstractPropagatorBuilder {
32  
33      /** Build a new instance.
34       * @param mu central attraction coefficient (m³/s²)
35       * @param frame the frame in which the orbit is propagated
36       * (<em>must</em> be a {@link Frame#isPseudoInertial pseudo-inertial frame})
37       * @deprecated as of 7.1, replaced with {@link #KeplerianPropagatorBuilder(double,
38       * Frame, OrbitType, PositionAngle)}
39       */
40      @Deprecated
41      public KeplerianPropagatorBuilder(final double mu, final Frame frame) {
42          this(mu, frame, OrbitType.KEPLERIAN, PositionAngle.TRUE);
43      }
44  
45      /** Build a new instance.
46       * @param mu central attraction coefficient (m³/s²)
47       * @param frame the frame in which the orbit is propagated
48       * (<em>must</em> be a {@link Frame#isPseudoInertial pseudo-inertial frame})
49       * @param orbitType orbit type to use
50       * @param positionAngle position angle type to use
51       * @since 7.1
52       */
53      public KeplerianPropagatorBuilder(final double mu, final Frame frame,
54                                        final OrbitType orbitType, final PositionAngle positionAngle) {
55          super(frame, mu, orbitType, positionAngle);
56      }
57  
58      /** {@inheritDoc} */
59      public Propagator buildPropagator(final AbsoluteDate date, final double[] parameters)
60          throws OrekitException {
61          checkParameters(parameters);
62          return new KeplerianPropagator(createInitialOrbit(date, parameters));
63      }
64  
65  }
