119     @Test
120     public void testOffsets() {
121 
122         // we arbitrary put UTC == TAI before 1961-01-01
123         checkOffset(1950,  1,  1,   0);
124 
125         // excerpt from UTC-TAI.history file:
126         //  1961  Jan.  1 - 1961  Aug.  1     1.422 818 0s + (MJD - 37 300) x 0.001 296s
127         //        Aug.  1 - 1962  Jan.  1     1.372 818 0s +        ""
128         //  1962  Jan.  1 - 1963  Nov.  1     1.845 858 0s + (MJD - 37 665) x 0.001 123 2s
129         //  1963  Nov.  1 - 1964  Jan.  1     1.945 858 0s +        ""
130         //  1964  Jan.  1 -       April 1     3.240 130 0s + (MJD - 38 761) x 0.001 296s
131         //        April 1 -       Sept. 1     3.340 130 0s +        ""
132         //        Sept. 1 - 1965  Jan.  1     3.440 130 0s +        ""
133         //  1965  Jan.  1 -       March 1     3.540 130 0s +        ""
134         //        March 1 -       Jul.  1     3.640 130 0s +        ""
135         //        Jul.  1 -       Sept. 1     3.740 130 0s +        ""
136         //        Sept. 1 - 1966  Jan.  1     3.840 130 0s +        ""
137         //  1966  Jan.  1 - 1968  Feb.  1     4.313 170 0s + (MJD - 39 126) x 0.002 592s
138         //  1968  Feb.  1 - 1972  Jan.  1     4.213 170 0s +        ""
139         checkOffset(1961,  1,  2,  -(1.422818 +   1 * 0.001296));  // MJD 37300 +   1
140         checkOffset(1961,  8,  2,  -(1.372818 + 213 * 0.001296));  // MJD 37300 + 213
141         checkOffset(1962,  1,  2,  -(1.845858 +   1 * 0.0011232)); // MJD 37665 +   1
142         checkOffset(1963, 11,  2,  -(1.945858 + 670 * 0.0011232)); // MJD 37665 + 670
143         checkOffset(1964,  1,  2,  -(3.240130 - 365 * 0.001296));  // MJD 38761 - 365
144         checkOffset(1964,  4,  2,  -(3.340130 - 274 * 0.001296));  // MJD 38761 - 274
145         checkOffset(1964,  9,  2,  -(3.440130 - 121 * 0.001296));  // MJD 38761 - 121
146         checkOffset(1965,  1,  2,  -(3.540130 +   1 * 0.001296));  // MJD 38761 +   1
147         checkOffset(1965,  3,  2,  -(3.640130 +  60 * 0.001296));  // MJD 38761 +  60
148         checkOffset(1965,  7,  2,  -(3.740130 + 182 * 0.001296));  // MJD 38761 + 182
149         checkOffset(1965,  9,  2,  -(3.840130 + 244 * 0.001296));  // MJD 38761 + 244
150         checkOffset(1966,  1,  2,  -(4.313170 +   1 * 0.002592));  // MJD 39126 +   1
151         checkOffset(1968,  2,  2,  -(4.213170 + 762 * 0.002592));  // MJD 39126 + 762
152 
153         // since 1972-01-01, offsets are only whole seconds
154         checkOffset(1972,  3,  5, -10);
155         checkOffset(1972,  7, 14, -11);
156         checkOffset(1979, 12, 31, -18);
157         checkOffset(1980,  1, 22, -19);
158         checkOffset(2006,  7,  7, -33);

167     @Test
168     public void testCreatingInLeapDate() {
169         AbsoluteDate previous = null;
170         final double step = 0.0625;
171         for (double seconds = 59.0; seconds < 61.0; seconds += step) {
172             final AbsoluteDate date = new AbsoluteDate(2008, 12, 31, 23, 59, seconds, utc);
173             if (previous != null) {
174                 Assert.assertEquals(step, date.durationFrom(previous), 1.0e-12);
175             }
176             previous = date;
177         }
178         AbsoluteDate ad0 = new AbsoluteDate("2008-12-31T23:59:60", utc);
179         Assert.assertTrue(ad0.toString(utc).startsWith("2008-12-31T23:59:"));
180         AbsoluteDate ad1 = new AbsoluteDate("2008-12-31T23:59:59", utc).shiftedBy(1);
181         Assert.assertEquals(0, ad1.durationFrom(ad0), 1.0e-15);
182         Assert.assertEquals(1, new AbsoluteDate("2009-01-01T00:00:00", utc).durationFrom(ad0), 1.0e-15);
183         Assert.assertEquals(2, new AbsoluteDate("2009-01-01T00:00:01", utc).durationFrom(ad0), 1.0e-15);
184     }

186     @Test
187     public void testDisplayDuringLeap() throws OrekitException {
188         AbsoluteDate t0 = utc.getLastKnownLeapSecond().shiftedBy(-1.0);
189         for (double dt = 0.0; dt < 3.0; dt += 0.375) {
190             AbsoluteDate t = t0.shiftedBy(dt);
191             double seconds = t.getComponents(utc).getTime().getSecond();
192             if (dt < 2.0) {
193                 Assert.assertEquals(dt + 59.0, seconds, 1.0e-12);
194             } else {
195                 Assert.assertEquals(dt - 2.0, seconds, 1.0e-12);
196             }
197         }

200     @Test
201     public void testMultithreading() {
202 
203         // generate reference offsets using a single thread
204         RandomGenerator random = new Well1024a(6392073424l);
205         List<AbsoluteDate> datesList = new ArrayList<AbsoluteDate>();
206         List<Double> offsetsList = new ArrayList<Double>();
207         AbsoluteDate reference = utc.getFirstKnownLeapSecond().shiftedBy(-Constants.JULIAN_YEAR);
208         double testRange = utc.getLastKnownLeapSecond().durationFrom(reference) + Constants.JULIAN_YEAR;
209         for (int i = 0; i < 10000; ++i) {
210             AbsoluteDate randomDate = reference.shiftedBy(random.nextDouble() * testRange);
211             datesList.add(randomDate);
212             offsetsList.add(utc.offsetFromTAI(randomDate));
213         }
214         
215         // check the offsets in multi-threaded mode
216         ExecutorService executorService = Executors.newFixedThreadPool(100);
217 
218         for (int i = 0; i < datesList.size(); ++i) {
219             final AbsoluteDate date = datesList.get(i);
220             final double offset = offsetsList.get(i);
221             executorService.execute(new Runnable() {
222                 public void run() {
223                     Assert.assertEquals(offset, utc.offsetFromTAI(date), 1.0e-12);
224                 }
225             });
226         }
227 
228         try {
229             executorService.shutdown();
230             executorService.awaitTermination(3, TimeUnit.SECONDS);
231         } catch (InterruptedException ie) {
232             Assert.fail(ie.getLocalizedMessage());
233         }
234 
235     }
