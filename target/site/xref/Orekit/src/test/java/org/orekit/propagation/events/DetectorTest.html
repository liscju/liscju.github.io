51      @Test
52      public void testBasicScheduling() throws OrekitException {
53  
54          final TimeScale utc = TimeScalesFactory.getUTC();
55          final Vector3D position = new Vector3D(-6142438.668, 3492467.56, -25767.257);
56          final Vector3D velocity = new Vector3D(505.848, 942.781, 7435.922);
57          final AbsoluteDate date = new AbsoluteDate(2003, 9, 16, utc);
58          final Orbit orbit = new CircularOrbit(new PVCoordinates(position,  velocity),
59                                                FramesFactory.getEME2000(), date, mu);
60  
61          Propagator propagator = new KeplerianPropagator(orbit);
62          double stepSize = 60.0;
63          OutOfOrderChecker checker = new OutOfOrderChecker(stepSize);
64          propagator.addEventDetector(new DateDetector(date.shiftedBy(5.25 * stepSize)).withHandler(checker));
65          propagator.setMasterMode(stepSize, checker);
66          propagator.propagate(date.shiftedBy(10 * stepSize));
67          Assert.assertTrue(checker.outOfOrderCallDetected());
68  
69      }

71      private static class OutOfOrderChecker implements EventHandler<DateDetector>, OrekitFixedStepHandler {
72  
73          private AbsoluteDate triggerDate;
74          private boolean outOfOrderCallDetected;
75          private double stepSize;
76  
77          public OutOfOrderChecker(final double stepSize) {
78              triggerDate = null;
79              outOfOrderCallDetected = false;
80              this.stepSize = stepSize;
81          }
82  
83          public Action eventOccurred(SpacecraftState s, DateDetector detector, boolean increasing) {
84              triggerDate = s.getDate();
85              return Action.CONTINUE;
86          }
87  
88          public SpacecraftState resetState(DateDetector detector, SpacecraftState oldState) {
89              return oldState;
90          }
91  
92          public void init(SpacecraftState s0, AbsoluteDate t) {
93          }
94  
95          public void handleStep(SpacecraftState currentState, boolean isLast) {
96              // step handling and event occurrences may be out of order up to one step
97              // with variable steps, and two steps with fixed steps (due to the delay
98              // induced by StepNormalizer)
99              if (triggerDate != null) {
100                 double dt = currentState.getDate().durationFrom(triggerDate);
101                 if (dt < 0) {
102                     outOfOrderCallDetected = true;
103                     Assert.assertTrue(FastMath.abs(dt) < (2 * stepSize));
104                 }
105             }
106         }
107 
108         public boolean outOfOrderCallDetected() {
109             return outOfOrderCallDetected;
110         }
111 
112     }

114     @Test
115     public void testIssue108Numerical() throws OrekitException {
116         final TimeScale utc = TimeScalesFactory.getUTC();
117         final Vector3D position = new Vector3D(-6142438.668, 3492467.56, -25767.257);
118         final Vector3D velocity = new Vector3D(505.848, 942.781, 7435.922);
119         final AbsoluteDate date = new AbsoluteDate(2003, 9, 16, utc);
120         final Orbit orbit = new CircularOrbit(new PVCoordinates(position,  velocity),
121                                               FramesFactory.getEME2000(), date, mu);
122         final double step = 60.0;
123         final int    n    = 100;
124         NumericalPropagator propagator = new NumericalPropagator(new ClassicalRungeKuttaIntegrator(step));
125         propagator.resetInitialState(new SpacecraftState(orbit));
126         GCallsCounter counter = new GCallsCounter(100000.0, 1.0e-6, 20, new StopOnEvent<DetectorTest.GCallsCounter>());
127         propagator.addEventDetector(counter);
128         propagator.propagate(date.shiftedBy(n * step));
129         Assert.assertEquals(n + 1, counter.getCount());
130     }

182     @Test
183     public void testNoisyGFunction() throws OrekitException {
184 
185         // initial conditions
186         Frame eme2000 = FramesFactory.getEME2000();
187         TimeScale utc = TimeScalesFactory.getUTC();
188         AbsoluteDate initialDate   = new AbsoluteDate(2011, 5, 11, utc);
189         AbsoluteDate startDate     = new AbsoluteDate(2032, 10, 17, utc);
190         AbsoluteDate interruptDate = new AbsoluteDate(2032, 10, 18, utc);
191         AbsoluteDate targetDate    = new AbsoluteDate(2211, 5, 11, utc);
192         KeplerianPropagator k1 =
193                 new KeplerianPropagator(new EquinoctialOrbit(new PVCoordinates(new Vector3D(4008462.4706055815, -3155502.5373837613, -5044275.9880020910),
194                                                                                new Vector3D(-5012.9298276860990, 1920.3567095973078, -5172.7403501801580)),
195                                                              eme2000, initialDate, Constants.WGS84_EARTH_MU));
196         KeplerianPropagator k2 =
197                 new KeplerianPropagator(new EquinoctialOrbit(new PVCoordinates(new Vector3D(4008912.4039522274, -3155453.3125615157, -5044297.6484738905),
198                                                                                new Vector3D(-5012.5883854112530, 1920.6332221785074, -5172.2177085540500)),
199                                                              eme2000, initialDate, Constants.WGS84_EARTH_MU));
200         k2.addEventDetector(new CloseApproachDetector(2015.243454166727, 0.0001, 100,
201                                                       new ContinueOnEvent<CloseApproachDetector>(),
202                                                       k1));
203         k2.addEventDetector(new DateDetector(Constants.JULIAN_DAY, 1.0e-6, interruptDate));
204         SpacecraftState s = k2.propagate(startDate, targetDate);
205         Assert.assertEquals(interruptDate, s.getDate());
206     }

208     private static class CloseApproachDetector extends AbstractDetector<CloseApproachDetector> {
209 
210         private static final long serialVersionUID = 1L;
211         private final PVCoordinatesProvider provider;
212 
213         public CloseApproachDetector(double maxCheck, double threshold,
214                                      final int maxIter, final EventHandler<CloseApproachDetector> handler,
215                                      PVCoordinatesProvider provider) {
216             super(maxCheck, threshold, maxIter, handler);
217             this.provider = provider;
218         }
219 
220         public double g(final SpacecraftState s) throws OrekitException {
221             PVCoordinates pv1     = provider.getPVCoordinates(s.getDate(), s.getFrame());
222             PVCoordinates pv2     = s.getPVCoordinates();
223             Vector3D deltaP       = pv1.getPosition().subtract(pv2.getPosition());
224             Vector3D deltaV       = pv1.getVelocity().subtract(pv2.getVelocity());
225             double radialVelocity = Vector3D.dotProduct(deltaP.normalize(), deltaV);
226             return radialVelocity;
227         }
228 
229         protected CloseApproachDetector create(final double newMaxCheck, final double newThreshold,
230                                                final int newMaxIter,
231                                                final EventHandler<CloseApproachDetector> newHandler) {
232             return new CloseApproachDetector(newMaxCheck, newThreshold, newMaxIter, newHandler,
233                                              provider);
234         }
235 
236     }
