174     private void checkPV(PVCoordinatesProvider body, AbsoluteDate date, Frame frame,
175                          Vector3D position, Vector3D velocity)
176     throws OrekitException {
177 
178         PVCoordinates pv = body.getPVCoordinates(date, frame);
179 
180         final double posScale = 149597870691.0;
181         final double velScale = posScale / Constants.JULIAN_DAY;
182         PVCoordinates reference =
183             new PVCoordinates(new Vector3D(posScale, position), new Vector3D(velScale, velocity));
184 
185         PVCoordinates error = new PVCoordinates(reference, pv);
186         Assert.assertEquals(0, error.getPosition().getNorm(), 2.0e-3);
187         Assert.assertEquals(0, error.getVelocity().getNorm(), 5.0e-10);
188 
189     }

191     @Test
192     public void testFrameShift() throws OrekitException {
193         Utils.setDataRoot("regular-data");
194         final Frame moon  = CelestialBodyFactory.getMoon().getBodyOrientedFrame();
195         final Frame earth = CelestialBodyFactory.getEarth().getBodyOrientedFrame();
196         final AbsoluteDate date0 = new AbsoluteDate(1969, 06, 25, TimeScalesFactory.getTDB());
197 
198         for (double t = 0; t < Constants.JULIAN_DAY; t += 3600) {
199             final AbsoluteDate date = date0.shiftedBy(t);
200             final Transform transform = earth.getTransformTo(moon, date);
201             for (double dt = -10; dt < 10; dt += 0.125) {
202                 final Transform shifted  = transform.shiftedBy(dt);
203                 final Transform computed = earth.getTransformTo(moon, transform.getDate().shiftedBy(dt));
204                 final Transform error    = new Transform(computed.getDate(), computed, shifted.getInverse());
205                 Assert.assertEquals(0.0, error.getTranslation().getNorm(),   100.0);
206                 Assert.assertEquals(0.0, error.getVelocity().getNorm(),       20.0);
207                 Assert.assertEquals(0.0, error.getRotation().getAngle(),    4.0e-8);
208                 Assert.assertEquals(0.0, error.getRotationRate().getNorm(), 8.0e-10);
209             }
210         }

213     @Test
214     public void testPropagationVsEphemeris() throws OrekitException {
215 
216         Utils.setDataRoot("regular-data");
217 
218         //Creation of the celestial bodies of the solar system
219         final CelestialBody sun     = CelestialBodyFactory.getSun();
220         final CelestialBody mercury = CelestialBodyFactory.getMercury();
221         final CelestialBody venus   = CelestialBodyFactory.getVenus();
222         final CelestialBody earth   = CelestialBodyFactory.getEarth();  
223         final CelestialBody mars    = CelestialBodyFactory.getMars();
224         final CelestialBody jupiter = CelestialBodyFactory.getJupiter();
225         final CelestialBody saturn  = CelestialBodyFactory.getSaturn();
226         final CelestialBody uranus  = CelestialBodyFactory.getUranus();
227         final CelestialBody neptune = CelestialBodyFactory.getNeptune();
228         final CelestialBody pluto   = CelestialBodyFactory.getPluto();
229 
230         //Starting and end dates
231         final AbsoluteDate startingDate = new AbsoluteDate(2000, 1, 2, TimeScalesFactory.getUTC());
232         AbsoluteDate endDate = startingDate.shiftedBy(30 * Constants.JULIAN_DAY);
233 
234         final Frame icrf = FramesFactory.getICRF();
235 
236         // fake orbit around negligible point mass at solar system barycenter
237         double negligibleMu = 1.0e-3;
238         SpacecraftState initialState = new SpacecraftState(new CartesianOrbit(venus.getPVCoordinates(startingDate, icrf),
239                                                            icrf, startingDate, negligibleMu));
240 
241         //Creation of the numerical propagator
242         final double[][] tol = NumericalPropagator.tolerances(1000, initialState.getOrbit(), OrbitType.CARTESIAN);
243         AbstractIntegrator dop1 = new DormandPrince853Integrator(1.0, 1.0e5, tol[0], tol[1]);
244         NumericalPropagator propag = new NumericalPropagator(dop1);
245         propag.setOrbitType(OrbitType.CARTESIAN);
246         propag.setInitialState(initialState);
247         propag.setMu(negligibleMu);
248 
249         //Creation of the ForceModels
250         propag.addForceModel(new BodyAttraction(sun)); 
251         propag.addForceModel(new BodyAttraction(mercury));
252         propag.addForceModel(new BodyAttraction(earth)); 
253         propag.addForceModel(new BodyAttraction(mars)); 
254         propag.addForceModel(new BodyAttraction(jupiter));
255         propag.addForceModel(new BodyAttraction(saturn));
256         propag.addForceModel(new BodyAttraction(uranus));
257         propag.addForceModel(new BodyAttraction(neptune));
258         propag.addForceModel(new BodyAttraction(pluto)); 
259 
260         // checks are done within the step handler
261         propag.setMasterMode(1000.0, new OrekitFixedStepHandler() {
262             public void init(SpacecraftState s0, AbsoluteDate t) {
263             }
264             public void handleStep(SpacecraftState currentState, boolean isLast)
265                 throws PropagationException {
266                 try {
267                     // propagated position should remain within 1400m of ephemeris for one month
268                     Vector3D propagatedP = currentState.getPVCoordinates(icrf).getPosition();
269                     Vector3D ephemerisP  = venus.getPVCoordinates(currentState.getDate(), icrf).getPosition();
270                     Assert.assertEquals(0, Vector3D.distance(propagatedP, ephemerisP), 1400.0);
271                 } catch (OrekitException oe) {
272                     throw new PropagationException(oe);
273                 }
274             }
275         });
276 
277         propag.propagate(startingDate,endDate);
278 
279     }

281     private static class BodyAttraction extends AbstractParameterizable implements ForceModel {
282 
283         /** Suffix for parameter name for attraction coefficient enabling jacobian processing. */
284         public static final String ATTRACTION_COEFFICIENT_SUFFIX = " attraction coefficient";
285 
286         /** The body to consider. */
287         private final CelestialBody body;
288 
289         /** Local value for body attraction coefficient. */
290         private double gm;
291 
292         /** Simple constructor.
293          * @param body the third body to consider
294          * (ex: {@link org.orekit.bodies.CelestialBodyFactory#getSun()} or
295          * {@link org.orekit.bodies.CelestialBodyFactory#getMoon()})
296          */
297         public BodyAttraction(final CelestialBody body) {
298             super(body.getName() + ATTRACTION_COEFFICIENT_SUFFIX);
299             this.body = body;
300             this.gm   = body.getGM();
301         }
302 
303         /** {@inheritDoc} */
304         public void addContribution(final SpacecraftState s, final TimeDerivativesEquations adder)
305             throws OrekitException {
306 
307             // compute bodies separation vectors and squared norm
308             final Vector3D centralToBody = body.getPVCoordinates(s.getDate(), s.getFrame()).getPosition();
309             final Vector3D satToBody     = centralToBody.subtract(s.getPVCoordinates().getPosition());
310             final double r2Sat           = satToBody.getNormSq();
311 
312             // compute relative acceleration
313             final Vector3D gamma =
314                 new Vector3D(gm / (r2Sat * FastMath.sqrt(r2Sat)), satToBody);
315 
316             // add contribution to the ODE second member
317             adder.addXYZAcceleration(gamma.getX(), gamma.getY(), gamma.getZ());
318 
319         }
320 
321         /** {@inheritDoc} */
322         public FieldVector3D<DerivativeStructure> accelerationDerivatives(final AbsoluteDate date, final Frame frame,
323                                                                           final FieldVector3D<DerivativeStructure> position,
324                                                                           final FieldVector3D<DerivativeStructure> velocity,
325                                                                           final FieldRotation<DerivativeStructure> rotation,
326                                                                           final DerivativeStructure mass)
327             throws OrekitException {
328 
329             // compute bodies separation vectors and squared norm
330             final Vector3D centralToBody    = body.getPVCoordinates(date, frame).getPosition();
331             final FieldVector3D<DerivativeStructure> satToBody = position.subtract(centralToBody).negate();
332             final DerivativeStructure r2Sat = satToBody.getNormSq();
333 
334             // compute absolute acceleration
335             return new FieldVector3D<DerivativeStructure>(r2Sat.pow(-1.5).multiply(gm), satToBody);
336 
337         }
338 
339         /** {@inheritDoc} */
340         public FieldVector3D<DerivativeStructure> accelerationDerivatives(final SpacecraftState s, final String paramName)
341             throws OrekitException {
342 
343             complainIfNotSupported(paramName);
344 
345             // compute bodies separation vectors and squared norm
346             final Vector3D centralToBody = body.getPVCoordinates(s.getDate(), s.getFrame()).getPosition();
347             final Vector3D satToBody     = centralToBody.subtract(s.getPVCoordinates().getPosition());
348             final double r2Sat           = Vector3D.dotProduct(satToBody, satToBody);
349 
350             final DerivativeStructure gmds = new DerivativeStructure(1, 1, 0, gm);
351 
352             // compute relative acceleration
353             return new FieldVector3D<DerivativeStructure>(gmds.multiply(FastMath.pow(r2Sat, -1.5)), satToBody);
354 
355         }
356 
357         /** {@inheritDoc} */
358         public EventDetector[] getEventsDetectors() {
359             return new EventDetector[0];
360         }
361 
362         /** {@inheritDoc} */
363         public double getParameter(final String name)
364             throws IllegalArgumentException {
365             complainIfNotSupported(name);
366             return gm;
367         }
368 
369         /** {@inheritDoc} */
370         public void setParameter(final String name, final double value)
371             throws IllegalArgumentException {
372             complainIfNotSupported(name);
373             gm = value;
374         }
375 
376     }

378     @Test
379     public void testKepler() throws OrekitException {
380         Utils.setDataRoot("regular-data");
381         AbsoluteDate date = new AbsoluteDate(1969, 06, 28, TimeScalesFactory.getTT());
382         final double au = 149597870691.0;
383         checkKepler(CelestialBodyFactory.getMoon(),    CelestialBodyFactory.getEarth(), date, 3.844e8, 0.012);
384         checkKepler(CelestialBodyFactory.getMercury(), CelestialBodyFactory.getSun(),   date,  0.387 * au, 4.0e-9);
385         checkKepler(CelestialBodyFactory.getVenus(),   CelestialBodyFactory.getSun(),   date,  0.723 * au, 8.0e-9);
386         checkKepler(CelestialBodyFactory.getEarth(),   CelestialBodyFactory.getSun(),   date,  1.000 * au, 2.0e-5);
387         checkKepler(CelestialBodyFactory.getMars(),    CelestialBodyFactory.getSun(),   date,  1.52  * au, 2.0e-7);
388         checkKepler(CelestialBodyFactory.getJupiter(), CelestialBodyFactory.getSun(),   date,  5.20  * au, 2.0e-6);
389         checkKepler(CelestialBodyFactory.getSaturn(),  CelestialBodyFactory.getSun(),   date,  9.58  * au, 8.0e-7);
390         checkKepler(CelestialBodyFactory.getUranus(),  CelestialBodyFactory.getSun(),   date, 19.20  * au, 6.0e-7);
391         checkKepler(CelestialBodyFactory.getNeptune(), CelestialBodyFactory.getSun(),   date, 30.05  * au, 4.0e-7);
392         checkKepler(CelestialBodyFactory.getPluto(),   CelestialBodyFactory.getSun(),   date, 39.24  * au, 3.0e-7);
393     }

395     private void checkKepler(final PVCoordinatesProvider orbiting, final CelestialBody central,
396                              final AbsoluteDate start, final double a, final double epsilon)
397         throws OrekitException {
398 
399         // set up Keplerian orbit of orbiting body around central body
400         Orbit orbit = new KeplerianOrbit(orbiting.getPVCoordinates(start, central.getInertiallyOrientedFrame()),
401                                          central.getInertiallyOrientedFrame(),start, central.getGM());
402         KeplerianPropagator propagator = new KeplerianPropagator(orbit);
403         Assert.assertEquals(a, orbit.getA(), 0.02 * a);
404         double duration = FastMath.min(50 * Constants.JULIAN_DAY, 0.01 * orbit.getKeplerianPeriod());
405 
406         double max = 0;
407         for (AbsoluteDate date = start; date.durationFrom(start) < duration; date = date.shiftedBy(duration / 100)) {
408             PVCoordinates ephemPV = orbiting.getPVCoordinates(date, central.getInertiallyOrientedFrame());
409             PVCoordinates keplerPV = propagator.propagate(date).getPVCoordinates();
410             Vector3D error = keplerPV.getPosition().subtract(ephemPV.getPosition());
411             max = FastMath.max(max, error.getNorm());
412         }
413         Assert.assertTrue(max < epsilon * a);
414     }
