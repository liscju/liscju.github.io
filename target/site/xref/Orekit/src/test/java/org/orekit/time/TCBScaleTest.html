14      @Test
15      public void testReference() {
16          TimeScale tcb = TimeScalesFactory.getTCB();
17          TimeScale tdb = TimeScalesFactory.getTDB();
18          Assert.assertEquals("TCB", tcb.toString());
19          AbsoluteDate refTCB = new AbsoluteDate("1977-01-01T00:00:32.184", tcb);
20          AbsoluteDate refTDB = new AbsoluteDate("1977-01-01T00:00:32.184", tdb);
21          Assert.assertEquals(0.0, refTCB.durationFrom(refTDB), 1.0e-12);
22      }

24      @Test
25      public void testRate() {
26          TimeScale tcb = TimeScalesFactory.getTCB();
27          TimeScale tdb = TimeScalesFactory.getTDB();
28          AbsoluteDate t0 = AbsoluteDate.J2000_EPOCH;
29          for (double deltaT = 1.0; deltaT < 10.0; deltaT += 0.3) {
30              AbsoluteDate t1 = t0.shiftedBy(deltaT);
31              double tdbRate = t1.offsetFrom(t0, tdb) / deltaT;
32              double tcbRate = t1.offsetFrom(t0, tcb) / deltaT;
33              Assert.assertEquals(tdbRate + 1.550519768e-8, tcbRate, 1.0e-14);
34          }
35      }

37      @Test
38      public void testSymmetry() {
39          TimeScale scale = TimeScalesFactory.getTCB();
40          for (double dt = -10000; dt < 10000; dt += 123.456789) {
41              AbsoluteDate date = AbsoluteDate.J2000_EPOCH.shiftedBy(dt * Constants.JULIAN_DAY);
42              double dt1 = scale.offsetFromTAI(date);
43              DateTimeComponents components = date.getComponents(scale);
44              double dt2 = scale.offsetToTAI(components.getDate(), components.getTime());
45              Assert.assertEquals( 0.0, dt1 + dt2, 1.0e-10);
46          }
47      }

49      @Test
50      public void testAAS06134() throws OrekitException {
51  
52          // this reference test has been extracted from the following paper:
53          // Implementation Issues Surrounding the New IAU Reference Systems for Astrodynamics
54          // David A. Vallado, John H. Seago, P. Kenneth Seidelmann
55          // http://www.centerforspace.com/downloads/files/pubs/AAS-06-134.pdf
56          // Note that the dUT1 here is -0.439962, whereas it is -0.4399619 in the book
57          Utils.setLoaders(IERSConventions.IERS_1996,
58                           Utils.buildEOPList(IERSConventions.IERS_1996, new double[][] {
59                               { 53098, -0.439962, 0.0015563, -0.140682, 0.333309, -0.052195, -0.003875, Double.NaN, Double.NaN },
60                               { 53099, -0.439962, 0.0015563, -0.140682, 0.333309, -0.052195, -0.003875, Double.NaN, Double.NaN },
61                               { 53100, -0.439962, 0.0015563, -0.140682, 0.333309, -0.052195, -0.003875, Double.NaN, Double.NaN },
62                               { 53101, -0.439962, 0.0015563, -0.140682, 0.333309, -0.052195, -0.003875, Double.NaN, Double.NaN },
63                               { 53102, -0.439962, 0.0015563, -0.140682, 0.333309, -0.052195, -0.003875, Double.NaN, Double.NaN },
64                               { 53103, -0.439962, 0.0015563, -0.140682, 0.333309, -0.052195, -0.003875, Double.NaN, Double.NaN },
65                               { 53104, -0.439962, 0.0015563, -0.140682, 0.333309, -0.052195, -0.003875, Double.NaN, Double.NaN },
66                               { 53105, -0.439962, 0.0015563, -0.140682, 0.333309, -0.052195, -0.003875, Double.NaN, Double.NaN }
67                           }));
68          AbsoluteDate date =
69                  new AbsoluteDate(2004, 4, 6, 7, 51, 28.386009, TimeScalesFactory.getUTC());
70          DateTimeComponents components = date.getComponents(TimeScalesFactory.getTCB());
71          Assert.assertEquals(2004,            components.getDate().getYear());
72          Assert.assertEquals(   4,            components.getDate().getMonth());
73          Assert.assertEquals(   6,            components.getDate().getDay());
74          Assert.assertEquals(   7,            components.getTime().getHour());
75          Assert.assertEquals(  52,            components.getTime().getMinute());
76  
77          // the "large" threshold in this test is due to the fact TDB model from which
78          // TCB is derived is approximated both in Orekit and in the reference paper.
79          // The difference is however small as the model in the paper is announced as
80          // being accurate to 50 micro seconds, and the test here is far below this value
81          Assert.assertEquals(  45.9109901113, components.getTime().getSecond(), 1.2e-8);
82  
83      }
