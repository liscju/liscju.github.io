118     @Test
119     public void testSerialization()
120             throws OrekitException, IOException, ClassNotFoundException {
121         for (Predefined predefined : Predefined.values()) {
122 
123             Frame original = FramesFactory.getFrame(predefined);
124             if (predefined == Predefined.ICRF) {
125                 Assert.assertEquals(CelestialBodyFactory.SOLAR_SYSTEM_BARYCENTER + "/inertial", original.getName());
126             } else {
127                 Assert.assertEquals(predefined.getName(), original.getName());
128             }
129 
130             ByteArrayOutputStream bos = new ByteArrayOutputStream();
131             ObjectOutputStream    oos = new ObjectOutputStream(bos);
132             oos.writeObject(original);
133              if (predefined == Predefined.GCRF) {
134                 Assert.assertTrue(bos.size() >  50);
135                 Assert.assertTrue(bos.size() < 100);
136             } else if (predefined == Predefined.ICRF) {
137                 Assert.assertTrue(bos.size() > 430);
138                 Assert.assertTrue(bos.size() < 480);
139             } else {
140                 Assert.assertTrue(bos.size() > 100);
141                 Assert.assertTrue(bos.size() < 160);
142             }
143 
144             ByteArrayInputStream  bis = new ByteArrayInputStream(bos.toByteArray());
145             ObjectInputStream     ois = new ObjectInputStream(bis);
146             Frame deserialized  = (Frame) ois.readObject();
147             Assert.assertTrue(original == deserialized);
148         }
149     }

151     @Test
152     public void testEOPConversionSymetry1980() throws OrekitException {
153         Utils.setDataRoot("rapid-data-columns");
154         IERSConventions.NutationCorrectionConverter converter =
155                 IERSConventions.IERS_1996.getNutationCorrectionConverter();

163             Assert.assertEquals(entry.getDdEps(), rebuiltEquinox[1], 2.0e-23);
164         }

193     private void testCIP(IERSConventions conventions, double threshold) throws OrekitException {
194         Utils.setLoaders(conventions, new ArrayList<EOPEntry>());
195         Frame cirf = FramesFactory.getCIRF(conventions, false);
196         Frame tod  = FramesFactory.getTOD(conventions, false);
197         AbsoluteDate t0 = new AbsoluteDate(new DateComponents(2003, 06, 21), TimeComponents.H00,
198                                            TimeScalesFactory.getUTC());
199         for (double dt = -30 * Constants.JULIAN_DAY; dt < 30 * Constants.JULIAN_DAY; dt += 3600) {
200             // CIRF and TOD should both have the Celestial Intermediate Pole as their Z axis
201             AbsoluteDate date = t0.shiftedBy(dt);
202             Transform t = FramesFactory.getNonInterpolatingTransform(tod, cirf, date);
203             Vector3D z = t.transformVector(Vector3D.PLUS_K);
204             Assert.assertEquals(0.0, Vector3D.angle(z, Vector3D.PLUS_K), threshold);
205         }
206     }

208     @Test
209     public void testEOPConversion() throws OrekitException {
210 
211         // real data from buletinb-298.txt
212 
213         // first use case: don't propagate the dx, dy correction to TOD, set dPsi, dEpsilon to 0.0
214         final List<EOPEntry> forced = Utils.buildEOPList(IERSConventions.IERS_2010, new double[][] {
215             { 56202, 0.3726886, 0.0008843, 0.168556, 0.332869, 0.0, 0.0, -0.000118, 0.000091 },
216             { 56203, 0.3719108, 0.0007204, 0.168261, 0.331527, 0.0, 0.0, -0.000140, 0.000111 },
217             { 56204, 0.3712561, 0.0006217, 0.168218, 0.330668, 0.0, 0.0, -0.000165, 0.000148 },
218             { 56205, 0.3706736, 0.0005530, 0.167775, 0.329688, 0.0, 0.0, -0.000188, 0.000189 },
219             { 56206, 0.3701593, 0.0005139, 0.166829, 0.328457, 0.0, 0.0, -0.000180, 0.000203 }
220         });
221 
222         Utils.setLoaders(IERSConventions.IERS_2010, forced);
223         Frame cirf            = FramesFactory.getCIRF(IERSConventions.IERS_2010, false);
224         Frame todNoCorrection = FramesFactory.getTOD(IERSConventions.IERS_2010, false);
225 
226         // second use case: convert dx, dy data into dDPsi, dDEpsilon
227         final List<EOPEntry> converted = Utils.buildEOPList(IERSConventions.IERS_2010, new double[][] {
228             { 56202, 0.3726886, 0.0008843, 0.168556, 0.332869, Double.NaN, Double.NaN, -0.000118, 0.000091 },
229             { 56203, 0.3719108, 0.0007204, 0.168261, 0.331527, Double.NaN, Double.NaN, -0.000140, 0.000111 },
230             { 56204, 0.3712561, 0.0006217, 0.168218, 0.330668, Double.NaN, Double.NaN, -0.000165, 0.000148 },
231             { 56205, 0.3706736, 0.0005530, 0.167775, 0.329688, Double.NaN, Double.NaN, -0.000188, 0.000189 },
232             { 56206, 0.3701593, 0.0005139, 0.166829, 0.328457, Double.NaN, Double.NaN, -0.000180, 0.000203 }
233         });
234         Utils.setLoaders(IERSConventions.IERS_2010, converted);
235         Frame todConvertedCorrection  = FramesFactory.getTOD(IERSConventions.IERS_2010, false);
236 
237         for (AbsoluteDate date = forced.get(0).getDate();
238              date.compareTo(forced.get(forced.size() - 1).getDate()) < 0;
239              date = date.shiftedBy(3600)) {
240             Transform tNoCorrection =
241                     FramesFactory.getNonInterpolatingTransform(todNoCorrection, cirf, date);
242 
243             // when we forget the correction on TOD,
244             // its Z axis is slightly offset from CIRF Z axis
245             Vector3D zNoCorrection  = tNoCorrection.transformVector(Vector3D.PLUS_K);
246             Assert.assertTrue(Vector3D.angle(zNoCorrection, Vector3D.PLUS_K) > 7.2e-10);
247             Transform tConverted =
248                     FramesFactory.getNonInterpolatingTransform(todConvertedCorrection, cirf, date);
249             // when we convert the correction and apply it to TOD,
250             // its Z axis is much better aligned with CIRF Z axis
251             Vector3D zConverted  = tConverted.transformVector(Vector3D.PLUS_K);
252             Assert.assertTrue(Vector3D.angle(zConverted, Vector3D.PLUS_K) < 6e-12);
253   
254         }
255 
256     }

258     @Test
259     public void testEOPConversionUAI2000Package() throws OrekitException {
260         // the reference value has been computed using the uai2000.package routines
261         // provided by Ch. Bizouard on page http://hpiers.obspm.fr/eop-pc/models/models_fr.html
262         // using the following main program
263         //
264         //        program test_EOP_conversion
265         //        double precision dmjd,dpsi,deps,dx1,dy1,dx2,dy2
266         //C       2004-02-14:00:00:00Z, MJD = 53049,
267         //C                             UT1-UTC=-0.4093475, LOD = 0.4676,
268         //C                             X = -0.076804, Y = 0.204671,
269         //C                             dx  = -0.075, dy  = -0.189
270         //C       values extracted from finals2000A.all file, bulletinA columns
271         //        dmjd = 53049.0
272         //        dx1  = -0.075
273         //        dy1  = -0.189
274         //        call DPSIDEPS2000_DXDY2000(dmjd,dx1,dy1,dpsi,deps)
275         //        write (6, *) 'dPsi = ', dpsi, 'dEpsilon = ', deps
276         //        call DXDY2000_DPSIDEPS2000(dmjd,dpsi,deps,dX2,dY2)
277         //        write (6, *) 'dx = ', dx2, 'dy = ', dy2
278         //      end
279         //
280         // the output of this test reads:
281         //     dPsi =  -0.18810999708158463      dEpsilon =  -0.18906891450729962
282         //     dx =   -7.5000002980232239E-002 dy =  -0.18899999558925629
283 
284         IERSConventions.NutationCorrectionConverter converter =
285                 IERSConventions.IERS_2003.getNutationCorrectionConverter();
286         AbsoluteDate date =
287                 new AbsoluteDate(new DateComponents(DateComponents.MODIFIED_JULIAN_EPOCH, 53049),
288                                  TimeScalesFactory.getUTC());
289         double dx  = Constants.ARC_SECONDS_TO_RADIANS * -0.075;
290         double dy  = Constants.ARC_SECONDS_TO_RADIANS * -0.189;
291         double[] equinox = converter.toEquinox(date, dx, dy);
292 
293         // The code from uai2000.package uses sin(epsilon0), cos(epsilon0) in the
294         // formula, whereas IERS conventions use sin(epsilonA), cos(epsilon0), i.e.
295         // the sine is computed on current date instead of epoch. This explains why
296         // the following threshold had to be raised to 2e-11.
297         // We still decided to stick with sin(epsilonA) in our implementation, in
298         // order to remain consistent with IERS conventions
299         Assert.assertEquals(Constants.ARC_SECONDS_TO_RADIANS * -0.18810999708158463,
300                             equinox[0], 2.0e-11);
301 
302         Assert.assertEquals(Constants.ARC_SECONDS_TO_RADIANS * -0.18906891450729962,
303                             equinox[1], 2.2e-14);
304         double[] nro = converter.toNonRotating(date, equinox[0], equinox[1]);
305         Assert.assertEquals(dx, nro[0], 1.0e-20);
306         Assert.assertEquals(dy, nro[1], 1.0e-20);
307 
308     }
