37      private static final ChronologicalComparator COMP = new ChronologicalComparator();
38  
39      @Test
40      public void testStartDateDaily1980() throws OrekitException, ParseException {
41          setRoot("rapid-data-xml");
42          IERSConventions.NutationCorrectionConverter converter =
43                  IERSConventions.IERS_1996.getNutationCorrectionConverter();
44          SortedSet<EOPEntry> history = new TreeSet<EOPEntry>(new ChronologicalComparator());
45          new RapidDataAndPredictionXMLLoader("^finals\\.daily\\.xml$").fillHistory(converter, history);
46          Assert.assertEquals(new AbsoluteDate(2010, 7, 1, TimeScalesFactory.getUTC()),
47                              new EOPHistory(IERSConventions.IERS_1996, history, true).getStartDate());
48      }
