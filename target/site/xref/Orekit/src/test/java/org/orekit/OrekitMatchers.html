41      /**
42       * Match a geodetic point
43       *
44       * @param lat latitude matcher, in radians
45       * @param lon longitude matcher, in radians
46       * @param alt altitude matcher, in meters
47       * @return a {@link GeodeticPoint} matcher
48       */
49      public static Matcher<GeodeticPoint> geodeticPoint(
50              final Matcher<Double> lat,
51              final Matcher<Double> lon,
52              final Matcher<Double> alt) {
53          return new TypeSafeDiagnosingMatcher<GeodeticPoint>() {
54              @Override
55              public void describeTo(Description description) {
56                  description.appendList("GeodeticPoint[", ",", "]",
57                          Arrays.<SelfDescribing>asList(lat, lon, alt));
58              }
59  
60              @Override
61              protected boolean matchesSafely(GeodeticPoint item,
62                                              Description mismatchDescription) {
63                  if (!lat.matches(item.getLatitude())) {
64                      mismatchDescription.appendText("the latitude ");
65                      lat.describeMismatch(item.getLatitude(),
66                              mismatchDescription);
67                      return false;
68                  }
69                  if (!lon.matches(item.getLongitude())) {
70                      mismatchDescription.appendText("the longitude ");
71                      lon.describeMismatch(item.getLongitude(),
72                              mismatchDescription);
73                      return false;
74                  }
75                  if (!alt.matches(item.getAltitude())) {
76                      mismatchDescription.appendText("the altitude ");
77                      alt.describeMismatch(item.getAltitude(),
78                              mismatchDescription);
79                      return false;
80                  }
81                  return true;
82              }
83          };
84      }

86      /**
87       * Match a geodetic point
88       *
89       * @param lat latitude, in radians
90       * @param lon longitude, in radians
91       * @param alt altitude, in meters
92       * @return matcher of a {@link GeodeticPoint}
93       */
94      public static Matcher<GeodeticPoint> geodeticPoint(double lat,
95                                                         double lon,
96                                                         double alt) {
97          return geodeticPoint(is(lat), is(lon), is(alt));
98      }

100     /**
101      * Match a geodetic point by comparing it with another one.
102      *
103      * @param expected the expected value
104      * @param absTol   the absolute tolerance on the comparison, in meters.
105      *                 Differences less than this value will be ignored.
106      * @return a {@link GeodeticPoint} matcher
107      */
108     public static Matcher<GeodeticPoint> geodeticPointCloseTo(
109             GeodeticPoint expected, double absTol) {
110         double angularAbsTol = absTol / Constants.WGS84_EARTH_EQUATORIAL_RADIUS;
111         return geodeticPoint(closeTo(expected.getLatitude(), angularAbsTol),
112                 closeTo(expected.getLongitude(), angularAbsTol),
113                 closeTo(expected.getAltitude(), absTol));
114     }

116     /**
117      * Match a geodetic point by comparing it with another one.
118      *
119      * @param expected the expected value
120      * @param ulps     the ulps difference allowed
121      * @return a {@link GeodeticPoint} matcher
122      * @see #relativelyCloseTo
123      */
124     public static Matcher<GeodeticPoint> geodeticPointCloseTo(
125             GeodeticPoint expected, int ulps) {
126         return geodeticPoint(relativelyCloseTo(expected.getLatitude(), ulps),
127                 relativelyCloseTo(expected.getLongitude(), ulps),
128                 relativelyCloseTo(expected.getAltitude(), ulps));
129     }

131     /**
132      * Matches a {@link Vector3D} based on its three coordinates.
133      *
134      * @param x matcher for the x coordinate
135      * @param y matcher for the y coordinate
136      * @param z matcher for the z coordinate
137      * @return a vector matcher
138      */
139     public static Matcher<Vector3D> vector(final Matcher<Double> x,
140                                            final Matcher<Double> y,
141                                            final Matcher<Double> z) {
142         return new TypeSafeDiagnosingMatcher<Vector3D>() {
143             @Override
144             public void describeTo(Description description) {
145                 description.appendList("Vector3D[", ",", "]",
146                         Arrays.<SelfDescribing>asList(x, y, z));
147             }
148 
149             @Override
150             protected boolean matchesSafely(Vector3D item,
151                                             Description mismatchDescription) {
152                 if (!x.matches(item.getX())) {
153                     mismatchDescription.appendText("the x coordinate ");
154                     x.describeMismatch(item.getX(), mismatchDescription);
155                     return false;
156                 }
157                 if (!y.matches(item.getY())) {
158                     mismatchDescription.appendText("the y coordinate ");
159                     y.describeMismatch(item.getY(), mismatchDescription);
160                     return false;
161                 }
162                 if (!z.matches(item.getZ())) {
163                     mismatchDescription.appendText("the z coordinate ");
164                     z.describeMismatch(item.getZ(), mismatchDescription);
165                     return false;
166                 }
167                 return true;
168             }
169         };
170     }

172     /**
173      * Matches a {@link Vector3D} close to another one.
174      *
175      * @param vector the reference vector
176      * @param absTol absolute tolerance of comparison, in each dimension
177      * @return a vector matcher.
178      */
179     public static Matcher<Vector3D> vectorCloseTo(Vector3D vector, double absTol) {
180         return vector(closeTo(vector.getX(), absTol),
181                 closeTo(vector.getY(), absTol), closeTo(vector.getZ(), absTol));
182     }

184     /**
185      * Matches a {@link Vector3D} close to another one.
186      *
187      * @param vector the reference vector
188      * @param ulps   the relative tolerance, in units in last place, of the
189      *               Comparison of each dimension.
190      * @return a vector matcher.
191      */
192     public static Matcher<Vector3D> vectorCloseTo(Vector3D vector, int ulps) {
193         return vector(relativelyCloseTo(vector.getX(), ulps),
194                 relativelyCloseTo(vector.getY(), ulps),
195                 relativelyCloseTo(vector.getZ(), ulps));
196     }

211     /**
212      * Matches a {@link Vector3D} to another one.
213      *
214      * @param vector the reference vector
215      * @param absTol the absolute tolerance of comparison, in each dimension.
216      * @param ulps   the relative tolerance of comparison in each dimension, in
217      *               units in last place.
218      * @return a matcher that matches if either the absolute or relative
219      * comparison matches in each dimension.
220      */
221     public static Matcher<Vector3D> vectorCloseTo(Vector3D vector,
222                                                   double absTol, int ulps) {
223         return vector(numberCloseTo(vector.getX(), absTol, ulps),
224                 numberCloseTo(vector.getY(), absTol, ulps),
225                 numberCloseTo(vector.getZ(), absTol, ulps));
226     }

228     /**
229      * Checks if two numbers are relatively close to each other. For absolute
230      * comparisons, see {@link #closeTo(double, double)}.
231      *
232      * @param expected the expected value in the relative comparison
233      * @param ulps     the units in last place of {@code expected} the two
234      *                 numbers can be off by.
235      * @return a matcher of numbers
236      */
237     public static Matcher<Double> relativelyCloseTo(final double expected,
238                                                     final int ulps) {
239         return new TypeSafeDiagnosingMatcher<Double>() {
240 
241             @Override
242             public void describeTo(Description description) {
243                 description.appendText("a numeric value within ")
244                         .appendValue(ulps).appendText(" ulps of ")
245                         .appendValue(expected);
246             }
247 
248             @Override
249             protected boolean matchesSafely(Double item,
250                                             Description mismatchDescription) {
251                 if (!Precision.equals(item, expected, ulps)) {
252                     mismatchDescription
253                             .appendValue(item)
254                             .appendText(" was off by ")
255                             .appendValue(
256                                     Double.doubleToLongBits(item)
257                                             - Double.doubleToLongBits(expected))
258                             .appendText(" ulps");
259                     return false;
260                 }
261                 return true;
262             }
263         };
264     }
265 
266     /**
267      * Check a number is close to another number using a relative
268      * <strong>or</strong> absolute comparison.
269      *
270      * @param number the expected value
271      * @param absTol absolute tolerance of comparison
272      * @param ulps   units in last place tolerance for relative comparison
273      * @return a matcher that matches if the differences is less than or equal
274      * to absTol <strong>or</strong> the two numbers differ by less or equal
275      * ulps.
276      */
277     public static Matcher<Double> numberCloseTo(double number, double absTol,
278                                                 int ulps) {
279         return either(closeTo(number, absTol)).or(
280                 relativelyCloseTo(number, ulps));
281     }

287     /**
288      * Creates a matcher of {@link Double}s that matches when an examined double
289      * is equal to the specified <code>operand</code>, within a range of +/-
290      * <code>error</code>. <p/> For example:
291      * <pre>assertThat(1.03, is(closeTo(1.0, 0.03)))</pre>
292      *
293      * @param value the expected value of matching doubles
294      * @param delta the delta (+/-) within which matches will be allowed
295      * @return a double matcher.
296      */
297     public static Matcher<Double> closeTo(final double value,
298                                           final double delta) {
299 
300         return new TypeSafeMatcher<Double>() {
301             @Override
302             public boolean matchesSafely(Double item) {
303                 return actualDelta(item) <= 0.0;
304             }
305 
306             @Override
307             public void describeMismatchSafely(Double item, Description mismatchDescription) {
308                 mismatchDescription.appendValue(item)
309                         .appendText(" differed by ")
310                         .appendValue(actualDelta(item));
311             }
312 
313             @Override
314             public void describeTo(Description description) {
315                 description.appendText("a numeric value within ")
316                         .appendValue(delta)
317                         .appendText(" of ")
318                         .appendValue(value);
319             }
320 
321             private double actualDelta(Double item) {
322                 return (Math.abs((item - value)) - delta);
323             }
324         };
