52      @Test
53      public void testParseOPM1()
54          throws OrekitException {
55          // simple test for OPM file, contains p/v entries and other mandatory
56          // data.
57          final String ex = "/ccsds/OPMExample.txt";
58  
59          final OPMParser parser = new OPMParser().withMu(398600e9).withConventions(IERSConventions.IERS_2010);
60          
61          final InputStream inEntry = getClass().getResourceAsStream(ex);
62  
63          final OPMFile file = parser.parse(inEntry, "OPMExample.txt");
64          final SatelliteTimeCoordinate coord = file.getSatelliteCoordinatesOPM();
65  
66          // Check Header Block;
67          Assert.assertEquals(2.0, file.getFormatVersion(), 1.0e-10);
68          Assert.assertEquals(new AbsoluteDate(1998, 11, 06, 9, 23, 57,
69                                               TimeScalesFactory.getUTC()), file
70              .getCreationDate());
71          Assert.assertEquals("JAXA", file.getOriginator());
72          
73          // Check Metadata Block;
74          
75          Assert.assertEquals("GODZILLA 5", file.getMetaData().getObjectName());
76          Assert.assertEquals("1998-057A", file.getMetaData().getObjectID());
77          Assert.assertEquals(1998, file.getMetaData().getLaunchYear());
78          Assert.assertEquals(57, file.getMetaData().getLaunchNumber());
79          Assert.assertEquals("A", file.getMetaData().getLaunchPiece());
80          Assert.assertEquals("EARTH", file.getMetaData().getCenterName());
81          Assert.assertTrue(file.getMetaData().getHasCreatableBody());
82          Assert.assertEquals(CelestialBodyFactory.getEarth(), file.getMetaData().getCenterBody());
83          Assert.assertEquals(CCSDSFrame.ITRF97.toString(), file.getMetaData().getFrame().getName());
84          Assert.assertEquals(TimeSystem.TAI, file.getTimeSystem());
85          
86          // Check State Vector data Block;
87          Assert.assertEquals(new AbsoluteDate(1998, 12, 18, 14, 28, 15.1172,
88                                               TimeScalesFactory.getTAI()), coord.getEpoch());
89          checkPVEntry(new PVCoordinates(new Vector3D(6503514.000, 1239647.000,
90                                                      -717490.000),
91                                         new Vector3D(-873.160, 8740.420,
92                                                      -4191.076)),
93                       coord.getCoordinate());
94          
95          
96          try {
97              file.generateCartesianOrbit();
98              Assert.fail("an exception should have been thrown");
99          } catch(IllegalArgumentException iae) {
100             // expected
101         }
102         try {
103             file.generateKeplerianOrbit();
104             Assert.fail("an exception should have been thrown");
105         } catch(IllegalArgumentException iae) {
106             // expected
107         }
108         try {
109             file.generateSpacecraftState();
110             Assert.fail("an exception should have been thrown");
111         } catch(IllegalArgumentException iae) {
112             // expected
113         }

117     @Test
118     public void testParseOPM2()
119         throws OrekitException {
120         // simple test for OPM file, contains all mandatory information plus
121         // Keplerian elements, Spacecraft parameters and 2 maneuvers.
122         final String ex = "/ccsds/OPMExample2.txt";
123 
124         final OPMParser parser = new OPMParser();
125         final InputStream inEntry = getClass().getResourceAsStream(ex);
126 
127         final OPMFile file = parser.parse(inEntry, "OPMExample2.txt");
128         final SatelliteTimeCoordinate coord = file.getSatelliteCoordinatesOPM();
129 
130         // Check Header Block;
131         Assert.assertEquals(2.0, file.getFormatVersion(), 1.0e-10);
132         ArrayList<String> headerComment = new ArrayList<String>();
133         headerComment.add("Generated by GSOC, R. Kiehling");
134         headerComment.add("Current intermediate orbit IO2 and maneuver planning data");
135         Assert.assertEquals(headerComment, file.getHeaderComment());
136         Assert.assertEquals(new AbsoluteDate(2000, 06, 03, 05, 33, 00,
137                                              TimeScalesFactory.getUTC()),
138                             file.getCreationDate());
139         Assert.assertEquals(file.getOriginator(), "GSOC");
140 
141         // Check Metadata Block;
142         
143         Assert.assertEquals("EUTELSAT W4", file.getMetaData().getObjectName());
144         Assert.assertEquals("2000-028A", file.getMetaData().getObjectID());
145         Assert.assertEquals("EARTH", file.getMetaData().getCenterName());
146         Assert.assertTrue(file.getMetaData().getHasCreatableBody());
147         Assert.assertEquals(CelestialBodyFactory.getEarth(), file.getMetaData().getCenterBody());
148         Assert.assertEquals(FramesFactory.getGCRF(), file.getMetaData().getFrame());
149         Assert.assertEquals(TimeSystem.GPS, file.getTimeSystem());
150 
151         // Check Data State Vector block
152         ArrayList<String> epochComment = new ArrayList<String>();
153         epochComment.add("State Vector");
154         Assert.assertEquals(epochComment, file.getEpochComment());
155         Assert.assertEquals(new AbsoluteDate(2006, 06, 03, 00, 00, 00,
156                                              TimeScalesFactory.getGPS()),
157                             coord.getEpoch());
158         checkPVEntry(new PVCoordinates(new Vector3D(6655994.2, -40218575.1, -82917.7),
159                                        new Vector3D(3115.48208, 470.42605, -1.01495)),
160                      coord.getCoordinate());
161 
162         // Check Data Keplerian Elements block
163         Assert.assertTrue(file.hasKeplerianElements());
164         ArrayList<String> keplerianElementsComment = new ArrayList<String>();
165         keplerianElementsComment.add("Keplerian elements");
166         Assert.assertEquals(keplerianElementsComment, file.getKeplerianElementsComment());
167         Assert.assertEquals(41399512.3, file.getA(), 1e-6);
168         Assert.assertEquals(0.020842611, file.getE(), 1e-10);
169         Assert.assertEquals(FastMath.toRadians(0.117746), file.getI(), 1e-10);
170         Assert.assertEquals(FastMath.toRadians(17.604721), file.getRaan(), 1e-10);
171         Assert.assertEquals(FastMath.toRadians(218.242943), file.getPa(), 1e-10);
172         Assert.assertEquals(PositionAngle.TRUE, file.getAnomalyType());
173         Assert.assertEquals(FastMath.toRadians(41.922339), file.getAnomaly(), 1e-10);
174         Assert.assertEquals(398600.4415 * 1e9, file.getMuParsed(), 1e-10);
175 
176         // Check Data Spacecraft block
177         ArrayList<String> spacecraftComment = new ArrayList<String>();
178         spacecraftComment.add("Spacecraft parameters");
179         Assert.assertEquals(spacecraftComment, file.getSpacecraftComment());        
180         Assert.assertEquals(1913.000, file.getMass(), 1e-10);
181         Assert.assertEquals(10.000, file.getSolarRadArea(), 1e-10);
182         Assert.assertEquals(1.300, file.getSolarRadCoeff(), 1e-10);
183         Assert.assertEquals(10.000, file.getDragArea(), 1e-10);
184         Assert.assertEquals(2.300, file.getDragCoeff(), 1e-10);
185 
186         // Check Data Maneuvers block
187         Assert.assertTrue(file.getHasManeuver());
188         Assert.assertEquals(3, file.getNbManeuvers());
189         ArrayList<String> stateManeuverComment0 = new ArrayList<String>();
190         stateManeuverComment0.add("2 planned maneuvers");
191         stateManeuverComment0.add("First maneuver: AMF-3");
192         stateManeuverComment0.add("Non-impulsive, thrust direction fixed in inertial frame");
193         Assert.assertEquals(stateManeuverComment0, file.getManeuver(0).getComment());
194         Assert.assertEquals(new AbsoluteDate(2000, 06, 03, 9, 00, 34.1,
195                                              TimeScalesFactory.getGPS()),
196                             file.getManeuvers().get(0).getEpochIgnition());
197         Assert.assertEquals(132.6, file.getManeuver(0).getDuration(),1e-10);
198         Assert.assertEquals(-18.418, file.getManeuver(0).getDeltaMass(), 1e-10);
199         Assert.assertNull(file.getManeuver(0).getRefLofType());
200         Assert.assertEquals(FramesFactory.getEME2000(), file.getManeuver(0).getRefFrame());
201         Assert.assertEquals(0.0,
202                             new Vector3D(-23.25700, 16.83160, -8.93444).distance(file.getManeuver(0).getDV()),
203                             1.0e-10);
204 
205         ArrayList<String> stateManeuverComment1 = new ArrayList<String>();
206         stateManeuverComment1.add("Second maneuver: first station acquisition maneuver");
207         stateManeuverComment1.add("impulsive, thrust direction fixed in RTN frame");
208         Assert.assertEquals(stateManeuverComment1, file.getManeuver(1).getComment());
209         Assert.assertEquals(new AbsoluteDate(2000, 06, 05, 18, 59, 21,
210                                              TimeScalesFactory.getGPS()),
211                             file.getManeuvers().get(1).getEpochIgnition());
212         Assert.assertEquals(0.0, file.getManeuver(1).getDuration(), 1e-10);
213         Assert.assertEquals(-1.469, file.getManeuver(1).getDeltaMass(), 1e-10);
214         Assert.assertEquals(LOFType.QSW, file.getManeuver(1).getRefLofType());
215         Assert.assertNull(file.getManeuver(1).getRefFrame());
216         Assert.assertEquals(0.0,
217                             new Vector3D(1.015, -1.873, 0.0).distance(file.getManeuver(1).getDV()),
218                             1.0e-10);
219 
220         Assert.assertTrue(file.getManeuver(2).getComment().isEmpty());
221         Assert.assertEquals(new AbsoluteDate(2000, 06, 05, 18, 59, 51,
222                                              TimeScalesFactory.getGPS()),
223                             file.getManeuvers().get(2).getEpochIgnition());
224         Assert.assertEquals(0.0, file.getManeuver(2).getDuration(), 1e-10);
225         Assert.assertEquals(-1.469, file.getManeuver(2).getDeltaMass(), 1e-10);
226         Assert.assertEquals(LOFType.QSW, file.getManeuver(2).getRefLofType());
227         Assert.assertNull(file.getManeuver(2).getRefFrame());
228         Assert.assertEquals(0.0,
229                             new Vector3D(1.015, -1.873, 0.0).distance(file.getManeuver(2).getDV()),
230                             1.0e-10);
231 
232         file.generateCartesianOrbit();
233         file.generateKeplerianOrbit();
234         file.generateSpacecraftState();
235 
236     }

252     @Test
253     public void testParseOPM3()
254             throws OrekitException, URISyntaxException {
255         // simple test for OPM file, contains all mandatory information plus
256         // Spacecraft parameters and the position/velocity Covariance Matrix.
257         final String name = getClass().getResource("/ccsds/OPMExample3.txt").toURI().getPath();
258         OPMParser parser = new OPMParser().withConventions(IERSConventions.IERS_2010);
259         final OPMFile file = parser.parse(name);
260         Assert.assertEquals(new AbsoluteDate(1998, 12, 18, 14, 28, 15.1172,
261                                              TimeScalesFactory.getGMST(IERSConventions.IERS_2010, false)),
262                             file.getMetaData().getFrameEpoch());
263         // Check Data Covariance matrix Block
264         ArrayList<String> dataCovMatrixComment = new ArrayList<String>();
265         dataCovMatrixComment.add("toto");
266         dataCovMatrixComment.add("tata");        
267         Assert.assertEquals(dataCovMatrixComment, file.getCovarianceComment());
268         Assert.assertTrue(file.hasCovarianceMatrix());
269         Assert.assertEquals(file.getCovRefFrame(), FramesFactory.getTEME());
270 
271         Array2DRowRealMatrix covMatrix = new Array2DRowRealMatrix(6, 6);
272         double[] column1 = {
273             3.331349476038534e-04, 4.618927349220216e-04,
274             -3.070007847730449e-04, -3.349365033922630e-07,
275             -2.211832501084875e-07, -3.041346050686871e-07
276         };
277         double[] column2 = {
278             4.618927349220216e-04, 6.782421679971363e-04,
279             -4.221234189514228e-04, -4.686084221046758e-07,
280             -2.864186892102733e-07, -4.989496988610662e-07
281         };
282         double[] column3 = {
283             -3.070007847730449e-04, -4.221234189514228e-04,
284             3.231931992380369e-04, 2.484949578400095e-07,
285             1.798098699846038e-07, 3.540310904497689e-07
286         };
287         double[] column4 = {
288             -3.349365033922630e-07, -4.686084221046758e-07,
289             2.484949578400095e-07, 4.296022805587290e-10,
290             2.608899201686016e-10, 1.869263192954590e-10
291         };
292         double[] column5 = {
293             -2.211832501084875e-07, -2.864186892102733e-07,
294             1.798098699846038e-07, 2.608899201686016e-10,
295             1.767514756338532e-10, 1.008862586240695e-10
296         };
297         double[] column6 = {
298             -3.041346050686871e-07, -4.989496988610662e-07,
299             3.540310904497689e-07, 1.869263192954590e-10,
300             1.008862586240695e-10, 6.224444338635500e-10
301         };
302         covMatrix.setColumn(0, column1);
303         covMatrix.setColumn(1, column2);
304         covMatrix.setColumn(2, column3);
305         covMatrix.setColumn(3, column4);
306         covMatrix.setColumn(4, column5);
307         covMatrix.setColumn(5, column6);
308         for (int i = 0; i < 6; i++) {
309             for (int j = 0; j < 6; j++) {
310                 Assert.assertEquals(covMatrix.getEntry(i, j), file
311                     .getCovarianceMatrix().getEntry(i, j), 1e-15);
312             }
313         }
314 
315         // Check User defined Parameters Block
316         HashMap<String, String> userDefinedParameters = new HashMap<String, String>();
317         userDefinedParameters.put("USER_DEFINED_EARTH_MODEL", "WGS-84");
318         userDefinedParameters.put("USER_DEFINED_TOTO", "TITI");
319         Assert.assertEquals(userDefinedParameters,
320                             file.getUserDefinedParameters());
321 
322     }

372         final String ex = "/ccsds/OPMExample4.txt";
373 
374         final OPMParser parser = new OPMParser().withMissionReferenceDate(new AbsoluteDate())
375                                                 .withConventions(IERSConventions.IERS_2010);
376 
377         final InputStream inEntry = getClass().getResourceAsStream(ex);
378         final OPMFile file = parser.parse(inEntry, "OPMExample4.txt");

395         try {
396             file.getEpochInterval();
397             Assert.fail("an exception should have been thrown");
398         } catch (UnsupportedOperationException uoe) {
399             // expected
400         }
401         try {
402             file.getNumberOfEpochs();
403             Assert.fail("an exception should have been thrown");
404         } catch (UnsupportedOperationException uoe) {
405             // expected
406         }

409     private void checkPVEntry(final PVCoordinates expected,
410                               final PVCoordinates actual) {
411         final Vector3D expectedPos = expected.getPosition();
412         final Vector3D expectedVel = expected.getVelocity();
413 
414         final Vector3D actualPos = actual.getPosition();
415         final Vector3D actualVel = actual.getVelocity();
416 
417         final double eps = 1e-12;
418 
419         Assert.assertEquals(expectedPos.getX(), actualPos.getX(), eps);
420         Assert.assertEquals(expectedPos.getY(), actualPos.getY(), eps);
421         Assert.assertEquals(expectedPos.getZ(), actualPos.getZ(), eps);
422 
423         Assert.assertEquals(expectedVel.getX(), actualVel.getX(), eps);
424         Assert.assertEquals(expectedVel.getY(), actualVel.getY(), eps);
425         Assert.assertEquals(expectedVel.getZ(), actualVel.getZ(), eps);
426     }
