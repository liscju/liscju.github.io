82          Orbit initialOrbit = new EquinoctialOrbit(new PVCoordinates(position, velocity),

84  
85          // Extrapolator definition
86          // -----------------------
87          KeplerianPropagator extrapolator = new KeplerianPropagator(initialOrbit);
88  
89          // Extrapolation at the initial date
90          // ---------------------------------
91          double delta_t = 0.0; // extrapolation duration in seconds
92          AbsoluteDate extrapDate = initDate.shiftedBy(delta_t);
93  
94          SpacecraftState finalOrbit = extrapolator.propagate(extrapDate);

98          double n = FastMath.sqrt(finalOrbit.getMu()/FastMath.pow(a, 3));

106         Assert.assertEquals(finalOrbit.getE(), initialOrbit.getE(), Utils.epsilonE * initialOrbit.getE());

146     @Test
147     public void propagatedCartesian() throws OrekitException {
148 
149         // Definition of initial conditions with position and velocity
150         //------------------------------------------------------------
151         Vector3D position = new Vector3D(7.0e6, 1.0e6, 4.0e6);
152         Vector3D velocity = new Vector3D(-500.0, 8000.0, 1000.0);
153         double mu = 3.9860047e14;
154 
155         AbsoluteDate initDate = AbsoluteDate.J2000_EPOCH.shiftedBy(584.);
156         Orbit initialOrbit = new EquinoctialOrbit(new PVCoordinates(position, velocity),
157                                                   FramesFactory.getEME2000(), initDate, mu);
158 
159         // Extrapolator definition
160         // -----------------------
161         KeplerianPropagator extrapolator = new KeplerianPropagator(initialOrbit);
162 
163         // Extrapolation at a final date different from initial date
164         // ---------------------------------------------------------
165         double delta_t = 100000.0; // extrapolation duration in seconds
166         AbsoluteDate extrapDate = initDate.shiftedBy(delta_t);
167 
168         SpacecraftState finalOrbit = extrapolator.propagate(extrapDate);
169 
170 
171         // computation of (M final - M initial) with another method
172         double a = finalOrbit.getA();
173         // another way to compute n
174         double n = FastMath.sqrt(finalOrbit.getMu()/FastMath.pow(a, 3));
175 
176         Assert.assertEquals(n * delta_t,
177                             finalOrbit.getLM() - initialOrbit.getLM(),
178                             Utils.epsilonAngle);
179 
180         // computation of M final orbit
181         double LM = finalOrbit.getLE()
182         - finalOrbit.getEquinoctialEx()*FastMath.sin(finalOrbit.getLE())
183         + finalOrbit.getEquinoctialEy()*FastMath.cos(finalOrbit.getLE());
184 
185         Assert.assertEquals(LM , finalOrbit.getLM() , Utils.epsilonAngle);
186 
187         // test of tan ((LE - Lv)/2) :
188         Assert.assertEquals(FastMath.tan((finalOrbit.getLE() - finalOrbit.getLv())/2.),
189                      tangLEmLv(finalOrbit.getLv(),finalOrbit.getEquinoctialEx(),finalOrbit.getEquinoctialEy()),
190                      Utils.epsilonAngle);
191 
192         // test of evolution of M vs E: LM = LE - ex*sin(LE) + ey*cos(LE)
193         // with ex and ey the same for initial and final orbit
194         double deltaM = finalOrbit.getLM() - initialOrbit.getLM();
195         double deltaE = finalOrbit.getLE() - initialOrbit.getLE();
196         double delta  = finalOrbit.getEquinoctialEx() * (FastMath.sin(finalOrbit.getLE()) - FastMath.sin(initialOrbit.getLE()))
197         - finalOrbit.getEquinoctialEy() * (FastMath.cos(finalOrbit.getLE()) - FastMath.cos(initialOrbit.getLE()));
198 
199         Assert.assertEquals(deltaM, deltaE - delta, Utils.epsilonAngle);
200 
201         // the orbital elements except for Mean/True/Eccentric latitude arguments are the same
202         Assert.assertEquals(finalOrbit.getA(), initialOrbit.getA(), Utils.epsilonTest * initialOrbit.getA());
203         Assert.assertEquals(finalOrbit.getEquinoctialEx(), initialOrbit.getEquinoctialEx(), Utils.epsilonE);
204         Assert.assertEquals(finalOrbit.getEquinoctialEy(), initialOrbit.getEquinoctialEy(), Utils.epsilonE);
205         Assert.assertEquals(finalOrbit.getHx(), initialOrbit.getHx(), Utils.epsilonAngle);
206         Assert.assertEquals(finalOrbit.getHy(), initialOrbit.getHy(), Utils.epsilonAngle);
207 
208         // for final orbit
209         double ex = finalOrbit.getEquinoctialEx();
210         double ey = finalOrbit.getEquinoctialEy();
211         double hx = finalOrbit.getHx();
212         double hy = finalOrbit.getHy();
213         double LE = finalOrbit.getLE();
214 
215         double ex2 = ex*ex;
216         double ey2 = ey*ey;
217         double hx2 = hx*hx;
218         double hy2 = hy*hy;
219         double h2p1 = 1. + hx2 + hy2;
220         double beta = 1. / (1. + FastMath.sqrt(1. - ex2 - ey2));
221 
222         double x3 = -ex + (1.- beta*ey2)*FastMath.cos(LE) + beta*ex*ey*FastMath.sin(LE);
223         double y3 = -ey + (1. -beta*ex2)*FastMath.sin(LE) + beta*ex*ey*FastMath.cos(LE);
224 
225         Vector3D U = new Vector3D((1. + hx2 - hy2)/ h2p1,
226                                   (2.*hx*hy)/h2p1,
227                                   (-2.*hy)/h2p1);
228 
229         Vector3D V = new Vector3D((2.*hx*hy)/ h2p1,
230                                   (1.- hx2+ hy2)/h2p1,
231                                   (2.*hx)/h2p1);
232 
233         Vector3D r = new Vector3D(finalOrbit.getA(),(new Vector3D(x3,U,y3,V)));
234 
235         Assert.assertEquals(finalOrbit.getPVCoordinates().getPosition().getNorm(), r.getNorm(), Utils.epsilonTest * r.getNorm());
236 
237     }

608         return (ey*FastMath.cos(Lv) - ex*FastMath.sin(Lv)) /
609         (1 + ex*FastMath.cos(Lv) + ey*FastMath.sin(Lv) + FastMath.sqrt(1 - ex*ex - ey*ey));
