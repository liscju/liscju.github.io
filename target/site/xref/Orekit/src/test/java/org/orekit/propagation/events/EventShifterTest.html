54      public void testNegNeg() throws OrekitException {
55          propagator.addEventDetector(createRawDetector("raw increasing", "raw decreasing", 1.0e-9));
56          propagator.addEventDetector(new EventShifter<EclipseDetector>(createRawDetector("shifted increasing", "shifted decreasing", 1.0e-3),

65          log.get(3).checkExpected(4361.986163327, "shifted increasing");
66          log.get(4).checkExpected(4376.986163327, "unshifted increasing");
67          log.get(5).checkExpected(4376.986163327, "raw increasing");
68      }

121     @Test
122     public void testIncreasingError() throws OrekitException {
123         propagator.addEventDetector(createRawDetector("raw increasing", "raw decreasing", 2.0e-9));
124         propagator.addEventDetector(new EventShifter<EclipseDetector>(createRawDetector("-10s increasing", "-10s decreasing", 2.0e-3),
125                                                                       true, -10, -10));
126         propagator.addEventDetector(new EventShifter<EclipseDetector>(createRawDetector("-100s increasing", "-100s decreasing", 3.0e-2),
127                                                                       true, -100, -100));
128         propagator.addEventDetector(new EventShifter<EclipseDetector>(createRawDetector("-1000s increasing", "-1000s decreasing", 5.0),
129                                                                       true, -1000, -1000));
130         propagator.propagate(iniDate.shiftedBy(20000));
131 
132         // the raw eclipses (not all within the propagation range) are at times:
133         // [ 2300.23843246594,   4376.986163326932]
134         // [ 8210.85851802963,  10287.572940950127]
135         // [14121.478252940502, 16198.159277277191]
136         // [20032.097637495113, 22108.745172638683]
137         // [25942.716671989547, 28019.330627364776]
138         // [31853.335356719457, 33929.91564178527]
139         // [ 37763.95369198012, 39840.50021622965]
140         Assert.assertEquals(26, log.size());
141         log.get( 0).checkExpected( 1300.238432465, "-1000s decreasing");
142         log.get( 1).checkExpected( 2200.238432465, "-100s decreasing");
143         log.get( 2).checkExpected( 2290.238432465, "-10s decreasing");
144         log.get( 3).checkExpected( 2300.238432465, "raw decreasing");
145         log.get( 4).checkExpected( 3376.986163327, "-1000s increasing");
146         log.get( 5).checkExpected( 4276.986163327, "-100s increasing");
147         log.get( 6).checkExpected( 4366.986163327, "-10s increasing");
148         log.get( 7).checkExpected( 4376.986163327, "raw increasing");
149         log.get( 8).checkExpected( 7210.858518030, "-1000s decreasing");
150         log.get( 9).checkExpected( 8110.858518030, "-100s decreasing");
151         log.get(10).checkExpected( 8200.858518030, "-10s decreasing");
152         log.get(11).checkExpected( 8210.858518030, "raw decreasing");
153         log.get(12).checkExpected( 9287.572940950, "-1000s increasing");
154         log.get(13).checkExpected(10187.572940950, "-100s increasing");
155         log.get(14).checkExpected(10277.572940950, "-10s increasing");
156         log.get(15).checkExpected(10287.572940950, "raw increasing");
157         log.get(16).checkExpected(13121.478252941, "-1000s decreasing");
158         log.get(17).checkExpected(14021.478252941, "-100s decreasing");
159         log.get(18).checkExpected(14111.478252941, "-10s decreasing");
160         log.get(19).checkExpected(14121.478252941, "raw decreasing");
161         log.get(20).checkExpected(15198.159277277, "-1000s increasing");
162         log.get(21).checkExpected(16098.159277277, "-100s increasing");
163         log.get(22).checkExpected(16188.159277277, "-10s increasing");
164         log.get(23).checkExpected(16198.159277277, "raw increasing");
165         log.get(24).checkExpected(19032.097637495, "-1000s decreasing");
166         log.get(25).checkExpected(19932.097637495, "-100s decreasing");
167 
168         for (EventEntry entry : log) {
169             double error = entry.getTimeError();
170             if (entry.name.contains("10s")) {
171                 Assert.assertTrue(error > 0.00001);
172                 Assert.assertTrue(error < 0.0003);
173             } else if (entry.name.contains("100s")) {
174                 Assert.assertTrue(error > 0.002);
175                 Assert.assertTrue(error < 0.03);
176             } else if (entry.name.contains("1000s")) {
177                 Assert.assertTrue(error > 0.7);
178                 Assert.assertTrue(error < 3.3);
179             }
180         }

183     private EclipseDetector createRawDetector(final String nameIncreasing, final String nameDecreasing,
184                                               final double tolerance)
185         throws OrekitException {
186         return new EclipseDetector(60., 1.e-10,
187                                    CelestialBodyFactory.getSun(), sunRadius,
188                                    CelestialBodyFactory.getEarth(), earthRadius).
189                                    withHandler(new EventHandler<EclipseDetector>() {
190                                        public Action eventOccurred(SpacecraftState s, EclipseDetector detector,
191                                                                    boolean increasing) {
192                                            log.add(new EventEntry(s.getDate().durationFrom(iniDate), tolerance,
193                                                                   increasing ? nameIncreasing : nameDecreasing));
194                                            return Action.CONTINUE;
195                                        }
196                                        public SpacecraftState resetState(EclipseDetector detector, SpacecraftState oldState) {
197                                            return oldState;
198                                        }
199                                    });
200     }

202     @Before
203     public void setUp() {
204         try {
205             Utils.setDataRoot("regular-data");
206             mu  = 3.9860047e14;
207             double ae  = 6.378137e6;
208             double c20 = -1.08263e-3;
209             double c30 = 2.54e-6;
210             double c40 = 1.62e-6;
211             double c50 = 2.3e-7;
212             double c60 = -5.5e-7;
213             final Vector3D position  = new Vector3D(-6142438.668, 3492467.560, -25767.25680);
214             final Vector3D velocity  = new Vector3D(505.8479685, 942.7809215, 7435.922231);
215             iniDate = new AbsoluteDate(1969, 7, 28, 4, 0, 0.0, TimeScalesFactory.getTT());
216             final Orbit orbit = new EquinoctialOrbit(new PVCoordinates(position,  velocity),
217                                                      FramesFactory.getGCRF(), iniDate, mu);
218             propagator =
219                 new EcksteinHechlerPropagator(orbit, ae, mu, c20, c30, c40, c50, c60);
220             log = new ArrayList<EventEntry>();
221         } catch (PropagationException pe) {
222             Assert.fail(pe.getLocalizedMessage());
223         }
224     }

233     private static class EventEntry {
234 
235         private final double dt;
236         private double expectedDT;
237         private final double tolerance;
238         private final String name;
239 
240         public EventEntry(final double dt, final double tolerance, final String name) {
241             this.dt         = dt;
242             this.expectedDT = Double.NaN;
243             this.tolerance  = tolerance;
244             this.name       = name;
245         }
246 
247         public void checkExpected(final double expectedDT, final String name) {
248             this.expectedDT = expectedDT;
249             Assert.assertEquals(expectedDT, dt, tolerance);
250             Assert.assertEquals(name, this.name);
251         }
252 
253         public double getTimeError() {
254             return FastMath.abs(dt - expectedDT);
255         }
