141     private void checkTilesDontOverlap(final List<List<Tile>> tiles) {
142         for (final List<Tile> list : tiles) {
143             for (final Tile tile : list) {
144                 final SphericalPolygonsSet quadrilateral =
145                         new SphericalPolygonsSet(1.0e-10,
146                                                  toS2Point(tile.getVertices()[0]),
147                                                  toS2Point(tile.getVertices()[1]),
148                                                  toS2Point(tile.getVertices()[2]),
149                                                  toS2Point(tile.getVertices()[3]));
150                 for (final List<Tile> otherList : tiles) {
151                     for (final Tile otherTile : otherList) {
152                         if (otherTile != tile) {
153                             for (final GeodeticPoint vertex : otherTile.getVertices()) {
154                                 Assert.assertEquals("tiles overlap at: " + vertex,
155                                                     Location.OUTSIDE, quadrilateral.checkPoint(toS2Point(vertex)));
156                             }
157                         }
158                     }
159                 }
160             }
161         }
162     }
163 
164     private S2Point toS2Point(final GeodeticPoint point) {
165         return new S2Point(point.getLongitude(), 0.5 * FastMath.PI - point.getLatitude());
166     }

218     private SphericalPolygonsSet buildSimpleZone(double[][] points) {
219         final S2Point[] vertices = new S2Point[points.length];
220         for (int i = 0; i < points.length; ++i) {
221             vertices[i] = new S2Point(FastMath.toRadians(points[i][1]),         // points[i][1] is longitude
222                                       FastMath.toRadians(90.0 - points[i][0])); // points[i][0] is latitude
223         }
224         return new SphericalPolygonsSet(1.0e-10, vertices);
225     }
