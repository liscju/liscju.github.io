236     private double[] derivativesErrors(PVCoordinatesProvider provider, AbsoluteDate date, Frame frame,
237                                        OneAxisEllipsoid model)
238         throws OrekitException {
239         List<TimeStampedPVCoordinates> pvList       = new ArrayList<TimeStampedPVCoordinates>();
240         List<TimeStampedPVCoordinates> groundPVList = new ArrayList<TimeStampedPVCoordinates>();
241         for (double dt = -0.25; dt <= 0.25; dt += 0.125) {
242             TimeStampedPVCoordinates shiftedPV = provider.getPVCoordinates(date.shiftedBy(dt), frame);
243             Vector3D p = model.projectToGround(shiftedPV.getPosition(), shiftedPV.getDate(), frame);
244             pvList.add(shiftedPV);
245             groundPVList.add(new TimeStampedPVCoordinates(shiftedPV.getDate(),
246                                                           p, Vector3D.ZERO, Vector3D.ZERO));
247         }
248         TimeStampedPVCoordinates computed =
249                 model.projectToGround(TimeStampedPVCoordinates.interpolate(date,
250                                                                            CartesianDerivativesFilter.USE_P,
251                                                                            pvList),
252                                                                            frame);
253         TimeStampedPVCoordinates reference =
254                 TimeStampedPVCoordinates.interpolate(date,
255                                                      CartesianDerivativesFilter.USE_P,
256                                                      groundPVList);
257 
258         TimeStampedPVCoordinates pv0 = provider.getPVCoordinates(date, frame);
259         Vector3D p0 = pv0.getPosition();
260         Vector3D v0 = pv0.getVelocity();
261         Vector3D a0 = pv0.getAcceleration();
262 
263         return new double[] {
264             Vector3D.distance(computed.getPosition(),     reference.getPosition())     / p0.getNorm(),
265             Vector3D.distance(computed.getVelocity(),     reference.getVelocity())     / v0.getNorm(),
266             Vector3D.distance(computed.getAcceleration(), reference.getAcceleration()) / a0.getNorm(),
267         };
268 
269     }

271     @Test
272     public void testGroundProjectionTaylor()
273             throws OrekitException {
274         Frame itrf = FramesFactory.getITRF(IERSConventions.IERS_2010, true);
275         Frame eme2000 = FramesFactory.getEME2000();
276         OneAxisEllipsoid model =
277             new OneAxisEllipsoid(Constants.WGS84_EARTH_EQUATORIAL_RADIUS,
278                                  Constants.WGS84_EARTH_FLATTENING,
279                                  itrf);
280 
281         TimeStampedPVCoordinates initPV =
282                 new TimeStampedPVCoordinates(AbsoluteDate.J2000_EPOCH.shiftedBy(584.),
283                                              new Vector3D(3220103., 69623., 6449822.),
284                                              new Vector3D(6414.7, -2006., -3180.),
285                                              Vector3D.ZERO);
286         Orbit orbit = new EquinoctialOrbit(initPV, eme2000, Constants.EIGEN5C_EARTH_MU);
287 
288         TimeStampedPVCoordinates pv0 = orbit.getPVCoordinates(orbit.getDate(), model.getBodyFrame());
289         PVCoordinatesProvider groundTaylor =
290                 model.projectToGround(pv0, model.getBodyFrame()).toTaylorProvider(model.getBodyFrame());
291 
292         TimeStampedPVCoordinates g0 = groundTaylor.getPVCoordinates(orbit.getDate(), model.getBodyFrame());
293         Vector3D zenith       = pv0.getPosition().subtract(g0.getPosition()).normalize();
294         Vector3D acrossTrack  = Vector3D.crossProduct(zenith, g0.getVelocity()).normalize();
295         Vector3D alongTrack   = Vector3D.crossProduct(acrossTrack, zenith).normalize();
296         for (double dt = -1; dt < 1; dt += 0.01) {
297             AbsoluteDate date = orbit.getDate().shiftedBy(dt);
298             Vector3D taylorP = groundTaylor.getPVCoordinates(date, model.getBodyFrame()).getPosition();
299             Vector3D refP    = model.projectToGround(orbit.getPVCoordinates(date, model.getBodyFrame()).getPosition(),
300                                                      date, model.getBodyFrame());
301             Vector3D delta = taylorP.subtract(refP);
302             Assert.assertEquals(0.0, Vector3D.dotProduct(delta, alongTrack),  0.0015);
303             Assert.assertEquals(0.0, Vector3D.dotProduct(delta, acrossTrack), 0.0007);
304             Assert.assertEquals(0.0, Vector3D.dotProduct(delta, zenith),      0.00002);
305         }
306 
307     }

439     public void testIntersectionFromPoints() throws OrekitException {
440         AbsoluteDate date = new AbsoluteDate(new DateComponents(2008, 03, 21),
441                                              TimeComponents.H12,
442                                              TimeScalesFactory.getUTC());
443 
444         Frame frame = FramesFactory.getITRF(IERSConventions.IERS_2010, true);
445         OneAxisEllipsoid earth = new OneAxisEllipsoid(6378136.460, 1 / 298.257222101, frame);
446 
447         // Satellite on polar position
448         // ***************************
449         final double mu = 3.9860047e14;
450         CircularOrbit circ =
451             new CircularOrbit(7178000.0, 0.5e-4, 0., FastMath.toRadians(90.), FastMath.toRadians(60.),
452                                    FastMath.toRadians(90.), PositionAngle.MEAN,
453                                    FramesFactory.getEME2000(), date, mu);
454 
455         // Transform satellite position to position/velocity parameters in EME2000 and ITRF200B
456         PVCoordinates pvSatEME2000 = circ.getPVCoordinates();
457         PVCoordinates pvSatItrf  = frame.getTransformTo(FramesFactory.getEME2000(), date).transformPVCoordinates(pvSatEME2000);
458         Vector3D pSatItrf  = pvSatItrf.getPosition();
459 
460         // Test first visible surface points
461         GeodeticPoint geoPoint = new GeodeticPoint(FastMath.toRadians(70.), FastMath.toRadians(60.), 0.);
462         Vector3D pointItrf     = earth.transform(geoPoint);
463         Line line = new Line(pSatItrf, pointItrf, 1.0e-10);
464         GeodeticPoint geoInter = earth.getIntersectionPoint(line, pSatItrf, frame, date);
465         Assert.assertEquals(geoPoint.getLongitude(), geoInter.getLongitude(), Utils.epsilonAngle);
466         Assert.assertEquals(geoPoint.getLatitude(), geoInter.getLatitude(), Utils.epsilonAngle);

568     private void checkCartesianToEllipsoidic(double ae, double f,
569                                              double x, double y, double z,
570                                              double longitude, double latitude,
571                                              double altitude)
572         throws OrekitException {
573 
574         AbsoluteDate date = AbsoluteDate.J2000_EPOCH;
575         Frame frame = FramesFactory.getITRF(IERSConventions.IERS_2010, true);
576         OneAxisEllipsoid model = new OneAxisEllipsoid(ae, f, frame);
577         GeodeticPoint gp = model.transform(new Vector3D(x, y, z), frame, date);
578         Assert.assertEquals(longitude, MathUtils.normalizeAngle(gp.getLongitude(), longitude), 1.0e-10);
579         Assert.assertEquals(latitude,  gp.getLatitude(),  1.0e-10);
580         Assert.assertEquals(altitude,  gp.getAltitude(),  1.0e-10 * FastMath.abs(ae));
581         Vector3D rebuiltNadir = Vector3D.crossProduct(gp.getSouth(), gp.getWest());
582         Assert.assertEquals(0, rebuiltNadir.subtract(gp.getNadir()).getNorm(), 1.0e-15);
583     }
