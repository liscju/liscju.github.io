11      @Test
12      public void recursionTest() throws OrekitException {
13  
14          for (int n = 2; n < 10; n++) {
15              final int mnm1 = -n - 1;
16              for (int s = 0; s < 10; s++) {
17                  final double newcomb10 = NewcombOperators.getValue(1, 0, mnm1, s);
18                  final double newcalc10 = (s - mnm1 / 2.);
19                  Assert.assertEquals(newcalc10, newcomb10, 0.);
20                  final double newcomb11 = NewcombOperators.getValue(1, 1, mnm1, s);
21                  final double newcalc11 = (-8 * s * s + 2 * mnm1 * mnm1 + 10 * mnm1 + 12) / 8.;
22                  Assert.assertEquals(newcalc11, newcomb11, 0.);
23                  final double newcomb20 = NewcombOperators.getValue(2, 0, mnm1, s);
24                  final double newcalc20 = (4. * s * s + 5 * s - 4 * mnm1 * s + mnm1 * mnm1 - 3 * mnm1) / 8.;
25                  Assert.assertEquals(newcalc20, newcomb20, 0.);
26                  final double newcomb02 = NewcombOperators.getValue(0, 2, mnm1, s);
27                  final double newcalc02 = (4. * s * s - 5 * s + 4 * mnm1 * s + mnm1 * mnm1 - 3 * mnm1) / 8.;
28                  Assert.assertEquals(newcalc02, newcomb02, 0.);
29                  final double newcomb21 = NewcombOperators.getValue(2, 1, mnm1, s);
30                  final double newcalc21 = -s*s*s/2. + (mnm1*s*s)/4. - (5.*s*s)/8. + (mnm1*mnm1*s)/8. + (21.*mnm1*s)/16. + (11*s)/8. - mnm1*mnm1*mnm1/16. - (7.*mnm1*mnm1)/16. - (9.*mnm1)/16.;
31                  Assert.assertEquals(newcalc21, newcomb21, 1.e-14);
32                  final double newcomb12 = NewcombOperators.getValue(1, 2, mnm1, s);
33                  final double newcalc12 =  s*s*s/2. + (mnm1*s*s)/4. - (5.*s*s)/8. - (mnm1*mnm1*s)/8. - (21.*mnm1*s)/16. - (11*s)/8. - mnm1*mnm1*mnm1/16. - (7.*mnm1*mnm1)/16. - (9.*mnm1)/16.;
34                  Assert.assertEquals(newcalc12, newcomb12, 1.e-14);
35                  final double newcomb22 = NewcombOperators.getValue(2, 2, mnm1, s);
36                  final double newcalc22 = s*s*s*s/4. - (mnm1*mnm1*s*s)/8. - mnm1*s*s - (105.*s*s)/64. + mnm1*mnm1*mnm1*mnm1/64. + (7.*mnm1*mnm1*mnm1)/32. + (71.*mnm1*mnm1)/64. + (77.*mnm1)/32. + 15./8.;
37                  Assert.assertEquals(newcalc22, newcomb22, 1.e-14);
38              }
39          }
40      }
